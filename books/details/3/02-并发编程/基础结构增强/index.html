实现一个并发安全的slice

有锁版本

import (
	"fmt"
	"strings"
	"sync"
	"sync/atomic"
)

type SafeSlice struct {
	data         []interface{}
	safeSliceLen int64
	mutex        *sync.Mutex
}

func NewSafeSlice() *SafeSlice {
	return &SafeSlice{
		data:  []interface{}{},
		mutex: &sync.Mutex{},
	}
}

func (s *SafeSlice) SafeSliceAppend(value interface{}) {
	s.mutex.Lock()
	s.data = append(s.data, value)
	s.mutex.Unlock()
	atomic.AddInt64(&s.safeSliceLen, 1)
}

无锁版本

  • 使用通道处理并发写
  • 使用链表引用避免读写之间的竞态
import (
	"fmt"
	"strings"
	"sync/atomic"
)

type SafeSliceByNode struct {
	head   *sliceNode
	tail   *sliceNode
	length int64
	ch     chan interface{} // 通道用来接收添加数据
}

type sliceNode struct {
	Val  interface{}
	Next *sliceNode
}

func NewSafeSliceByNode() *SafeSliceByNode {
	s := &SafeSliceByNode{
		head:   nil,
		tail:   nil,
		length: 0,
		ch:     make(chan interface{}, 100),
	}
	go func() {
		for item := range s.ch {
			cur := &sliceNode{
				Val: item,
			}
			if atomic.LoadInt64(&s.length) == 0 {
				s.head = cur
				s.tail = cur
			} else {
				s.tail.Next = cur
				s.tail = cur
			}
			atomic.AddInt64(&s.length, 1)
		}
	}()

	return s
}

// 基于channel
func (s *SafeSliceByNode) SafeSliceAppend(value interface{}) {
	s.ch <- value
}

func (s *SafeSliceByNode) SafeSliceRead() string {
	length := atomic.LoadInt64(&s.length)
	res := []string{}

	var cur *sliceNode
	for i := 0; i < int(length); i++ {
		if i == 0 {
			cur = s.head
		} else {
			cur = cur.Next
		}
		res = append(res, fmt.Sprint(cur.Val))
	}
	return strings.Join(res, ",")
}

// 清理操作
func (s *SafeSliceByNode) Close() {
	// 暂时不需要等待没有加载完的，因为程序本来就要关闭了
	close(s.ch)
}
