获取上层函数调用名称

func FnName(bottom int) string {
	pc, _, _, ok := runtime.Caller(bottom)
	if !ok {
		panic("not found caller")
	}
	fn := runtime.FuncForPC(pc)
	name := fn.Name()
	return name
}
