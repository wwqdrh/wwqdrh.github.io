通过golang自带的trace工具进行分析

下面是一个示例

import (
	"fmt"
	"os"
	"runtime"
	"runtime/trace"
	"time"
)

func numFinizaler() {
	var i = 3
	// 这里的func 就是后面一直说的 finalizer
	runtime.SetFinalizer(&i, func(i *int) {
		fmt.Println(i, *i, "set finalizer")
	})
}

func main() {
	f, _ := os.Create("trace.out")
	defer f.Close()
	trace.Start(f)
	defer trace.Stop()

	numFinizaler()
	runtime.GC()
	time.Sleep(3 * time.Second)
	runtime.GC()
}

go run ..

go tool trace trace.out
