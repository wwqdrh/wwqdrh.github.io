context

  • context.WithTimeout: 相较于当前时间
  • context.WithDeadline: 设定过期时间
  • context.WithValue: 设置上下文值
ctx = context.WithValue(ctx, timeoutKey, "timeout exceeded")
ctx = context.WithValue(ctx, deadlineKey, "deadline exceeded")

timeoutCtx, cancel := context.WithTimeout(ctx, (time.Duration(rand.Intn(2)) * time.Millisecond))
defer cancel()

deadlineCtx, cancel := context.WithDeadline(ctx, time.Now().Add(time.Duration(rand.Intn(2))*time.Millisecond))
defer cancel()

// 查询value的时候会向上查询
for {
    select {
    case <-timeoutCtx.Done():
        fmt.Println(GetValue(ctx, timeoutKey))
        return
    case <-deadlineCtx.Done():
        fmt.Println(GetValue(ctx, deadlineKey))
        return
    }
}

设计一个goroutine local


import (
	"bytes"
	"encoding/json"
	"errors"
	"fmt"
	"runtime"
	"strconv"
	"sync"
)

type localValue = *sync.Map // 每个协程存储值的位置

var (
	goroutineSpace = []byte("goroutine ")

	goroutineLocal = sync.Map{} // 协程id与上下文值的存储
)

// GoroutineID 获取当前goroutine的id
func GoroutineID() uint64 {
	b := make([]byte, 64)
	b = b[:runtime.Stack(b, false)]
	// Parse the 4707 out of "goroutine 4707 ["
	b = bytes.TrimPrefix(b, goroutineSpace)
	i := bytes.IndexByte(b, ' ')
	if i < 0 {
		panic(fmt.Sprintf("No space found in %q", b))
	}
	b = b[:i]
	n, err := strconv.ParseUint(string(b), 10, 64)
	if err != nil {
		panic(fmt.Sprintf("Failed to parse goroutine ID out of %q: %v", b, err))
	}
	return n
}

func HasContext() bool {
	id := GoroutineID()
	_, ok := goroutineLocal.Load(id)
	return ok
}

func ClearContext() {
	id := GoroutineID()
	goroutineLocal.Delete(id)
}

func GetContextValue(key string) interface{} {
	id := GoroutineID()
	val, ok := goroutineLocal.Load(id)
	if !ok {
		return nil
	}

	ctx, ok := val.(localValue)
	if !ok {
		return nil
	}

	if val, ok := ctx.Load(key); !ok {
		return nil
	} else {
		return val
	}
}

// 返回当前上下文中所有数据用字符串返回
func GetContextJson() string {
	id := GoroutineID()
	val, ok := goroutineLocal.Load(id)
	if !ok {
		return ""
	}

	ctx, ok := val.(localValue)
	if !ok {
		return ""
	}

	values := map[string]interface{}{}
	ctx.Range(func(key, value interface{}) bool {
		values[key.(string)] = value
		return true
	})
	bydata, err := json.Marshal(values)
	if err != nil {
		return ""
	}
	return string(bydata)
}

func SetContextValue(key string, value interface{}) error {
	id := GoroutineID()
	val, _ := goroutineLocal.LoadOrStore(id, &sync.Map{})

	ctx, ok := val.(localValue)
	if !ok {
		return errors.New("localvalue格式错误")
	}

	ctx.Store(key, value)
	return nil
}
