# 自定义规则

## 累加数

判断一个字符串是否满足a+b=c, ... 累加的规则

整个累加数只由第一个和第二个决定，穷举第一个和第二个字符串的所有可能性。

```go
func isAdditiveNumber(num string) bool {
	n := len(num)
	for secondStart := 1; secondStart < n - 1; secondStart++ {
		if num[0] == '0' && secondStart != 1 {
			break
		}
		for secondEnd := secondStart; secondEnd < n - 1; secondEnd++ {
			if num[secondStart] == '0' && secondStart != secondEnd {
				break
			}
			if valid(num, secondStart, secondEnd) {
				return true
			}
		}
	}
	return false
}

func valid(num string, secondStart, secondEnd int) bool {
	n := len(num)
	firstStart, firstEnd := 0, secondStart - 1
	for secondEnd <= n - 1 {
		third := stringAdd(num[firstStart:firstEnd+1], num[secondStart:secondEnd+1])
		
		thirdStart := secondEnd + 1
		thirdEnd := secondEnd + len(third)
		if thirdEnd >= n || num[thirdStart:thirdEnd+1] != third {
			break
		}
		if thirdEnd == n - 1 {
			return true
		}
		// 迭代
		firstStart, firstEnd = secondStart, secondEnd
		secondStart, secondEnd = thirdStart, thirdEnd
	}
	return false
}

func stringAdd(x, y string) string {
	res := []byte{}
	carry, cur := 0, 0 // 进位，当前值
	for x != "" || y != "" || carry != 0 {
		cur = carry
		if x != "" {
			cur += int(x[len(x)-1] - '0')
			x = x[:len(x)-1]
		}
		if y != "" {
			cur += int(y[len(y)-1] - '0')
			y = y[:len(y)-1]
		}
		carry = cur / 10
		cur %= 10
		res = append(res, byte(cur)+'0')
	}
	for i, n := 0, len(res); i < n/2; i++ {
		res[i], res[n-1-i] = res[n-1-i], res[i]
	}
	return string(res)
}
```

```go
func countVowelPermutation(n int) (ans int) {
	const mod int = 1e9 + 7
	dp := [5]int{1,1,1,1,1} // 状态优化，每个种类只与前面的一排有关
	for i := 1; i < n; i++ {
		dp = [5]int{
			(dp[1] + dp[2] + dp[4]) % mod, // a的前面可以为e、u、i
			(dp[0] + dp[2]) % mod, // e前面可以为a、i
			(dp[1] + dp[3]) % mod, // i前面可以为e、o
			dp[2], // o前面可以为i
			(dp[2] + dp[3]) % mod, // u前面可以为i、o
		}
	}

		ans = (ans + v) % mod
	}
	return
}
```

## 通过一系列规则构造出字典序字符串

`words = ["wrt", "wrf", "er", "ett", "rftt"]`

需要寻找到满足规则的字典序

需要注意的是字典顺序不是单个`words[i]`决定，而是由words中的所有字符串来决定的顺序

#拓扑排序

将每个字母看做节点，现在给出节点中的连接方向，需要寻找到一个最短路径将所有的节点连接起来(即拓扑排序)。需要注意的是这里有向图，可能存在环路

```go
func alienOrder(words []string) string {
	g := map[byte][]byte{}  // 正向邻接表
	inDeg := map[byte]int{} // 入度表

	// 构建入度表以及邻接表
	for _, c := range words[0] {
		inDeg[byte(c)] = 0
	}
next:
	for i := 1; i < len(words); i++ {
		pre, cur := words[i-1], words[i]
		for _, c := range cur {
			inDeg[byte(c)] += 0 // 为了让map中存在这个`byte(c)`
		}
		for j := 0; j < len(pre) && j < len(cur); j++ {
			if pre[j] != cur[j] {
				// 不相等，说明前面的比后面的小
				g[pre[j]] = append(g[pre[j]], cur[j])
				inDeg[cur[j]]++
				continue next
			}
		}
		if len(pre) > len(cur) {
			return ""
		}
	}

	// 拓扑排序
	res := make([]byte, len(inDeg))
	q := res[:0]
	for ch, num := range inDeg {
		if num == 0 {
			q = append(q, ch)
		}
	}
	for len(q) > 0 {
		u := q[0]
		q = q[1:]
		for _, v := range g[u] {
			if inDeg[v]--; inDeg[v] == 0 {
				q = append(q, v)
			}
		}
	}
	if cap(q) == 0 {
		// 容量减少是因为上文中的q = q[1:]
		// 这么判断是因为
		// 最初容量是由indeg决定的，即元素个数
		// 当最终q容量为0的时候，即所有的元素都已经参加拓扑排序了
		return string(res)
	}
	return ""
}
```

## 按照给定的字符串字典，判断能否构造出target字符串

#状态压缩DP

设target的长度为m，一共有2^m个子序列

拼出某个子序列mask所需要的最小贴纸数可以由mask的子序列来决定，满足动态规划思路。设dp(mask)为不同状态的最小贴纸数

计算dp(mask)需要使用最优的sticker来贡献部分字符，选取最优的算法：`遍历所有sticker，计算mask与sticker的最大交集，剩余部分由动态规划继续计算`，这部分可以使用状压+位运算

```go
func minStickers(stickers []string, target string) int {
	m := len(target)       // target的长度，由于构造2^m子序列
	f := make([]int, 1<<m) // dp, f[i]表示i状态下所需要的最少贴纸数
	for i := range f {
		f[i] = -1
	}
	f[0] = 0

	var dp func(int) int
	dp = func(mask int) int {
		if f[mask] != -1 {
			return f[mask]
		}

		f[mask] = m + 1 // 默认的数量大于m，如果未找到最后会返回m+1
		for _, sticker := range stickers {
			left := mask // 剩下的元素，继续dp

			// 记录当前sticker的元素个数
			cnt := [26]int{}
			for _, c := range sticker {
				cnt[c-'a']++
			}
			for i, c := range target {
				if mask>>i&1 == 1 && cnt[c-'a'] > 0 {
					// 判断最低位、是否选择
					cnt[c-'a']--
					left ^= 1 << i // 异或, 1<<i这个位置，原来是1->0(这里一定是1，因为由mask决定的，mask>>i&1为1才会进来)
				}
			}

			// left变成了减去ticker的元素
			if left < mask {
				f[mask] = min(f[mask], dp(left)+1)
			}
		}
		return f[mask]
	}
	ans := dp(1<<m - 1) // 1<<m - 1表示状态位上的所有元素都选择了
	if ans <= m {
		return ans
	}
	// 如果返回的m+1(dp函数里面定义的)，说明没有找到能够组成target的sticker，返回-1
	return -1
}

func min(a, b int) int {
	if a > b {
		return b
	}
	return a
}
```

## 标签验证器

验证字符串是否合法

- `<tag></tag>`标签必须闭合，tag必须大写且长度为1~9
- `<![CDATA[CDATA_CONTENT]]>`, CDATA_CONTENT中的数据不做校验

#栈

```go
func isValid(code string) bool {
	tags := []string{} // 栈
	for code != "" {
		if code[0] != '<' {
			if len(tags) == 0 {
				// 说明当前的数据不在tag中
				return false
			}
			code = code[1:]
			continue
		}
		if len(code) == 1 {
			// 只有<
			return false
		}
		
		if code[1] == '/' {
			// 进入检验闭合标签的逻辑
			j := strings.IndexByte(code, '>')
			if j == -1 {
				return false
			}
			tags = tags[:len(tags)-1] //将开始标签弹出 
			code = code[j+1:]
			if len(tags) == 0 && code != "" {
				// 没有标签且还存在字符串
				return false
			}
		} else if code[1] == '!' {
			// 进入CDATA校验逻辑
			if len(tags) == 0 || len(code) < 9 || code[2:9] != "[CDATA[" {
				// CDATA校验失败
				return false
			}
			j := strings.Index(code, "]]>")
			if j == -1 {
				return false
			}
			code = code[j+3:] // CDATA中的数据不校验直接跳过
		} else {
			// 起始标签校验逻辑
			j := strings.Index.Byte(code, '>')
			if j == -1 {
				return false
			}
			tagName := code[1:j]
			if tagName == "" || len(tagName) > 9 {
				return false
			}
			for _, ch := range tagName {
				if !unicode.IsUpper(ch) {
					return false
				}
			}
			tags = append(tags, tagName)
			code = code[j+1:]
		}
	}
	return len(tags) == 0
}
```

## 多对一

### 多个字符串排列组合构成一个匹配串，一个目标字符串

#滑动窗口

将目标字符串分组后，进行滑动窗口遍历，维护目标字符串组中与匹配串之间的频率差值，如果所有差值为0那么就是匹配成功了，当期的左边界是结果

```go
// words中的每个字符串长度是相同的
func findSubstring(s string, words []string) (ans []int) {
	ls, m, n := len(s), len(words), len(words[0])
	for i := 0; i < n && i+m*n <= ls; i++ {
		// 给匹配字符串分组一共有n-1种分组
		differ := map[string]int{} // 记录频率差值
		for j := 0; j < m; j++ {
			// 当前分组字符串情况
			differ[s[i+j*n: i+(j+1)*n]]++
		}
		for _, word := range words {
			differ[word]--
			if differ[word] == 0 {
				delete(differ, word)
			}
		}
		// 滑动窗口 左边的减少即differ减少 右边的增加即differ增加
		for start := i; start < ls-m*n+1; start += n {
			if start != i {
				word := s[start+(m-1)*n: start+m*n] // 右边的元素
				differ[word]++
				if differ[word] == 0 {
					delete(differ, word)
				}
				word = s[start-n: start] // 左边的元素
				differ[word]--
				if differ[word] == 0 {
					delete(differ, word)
				}
			}
			if len(differ) == 0 {
				ans = append(ans, start)
			}
		}
	}
	return
}
```

## 一对一(一个匹配串, 一个目标字符串)

> 暴力匹配O(mn)是时间复杂度

## Rabin-Karp

利用hash函数快速判断两个是否能组成成功，每次不用重新计算整个字符串的hash而是直接`原hash值 * R + s[k-1] - s[i]*R^(k-1)`

时间复杂度: O(m+n)
最坏时间复杂度：O(mn)

```Go
const primeRK = 16777619 // 32位FNV hash算法中的基础质数（相当于进制）

// hashStr 散列函数 返回字符串以及primeRK的k-1 ^ (len(sep)-1)
func hashStr(sep string) (uint32, uint32) {
	// 字符串hash
	var hash uint32 = 0
	for i := 0; i < len(sep); i++ {
		hash = hash*primeRK + uint32(sep[i]) // 循环得到字符串hash
	}

	// 获取primeRK 的 len(sep)-1 次方 快速幂快速求平方
	var pow, sq uint32 = 1, primeRK
	for i := len(sep); i > 0; i >>= 1 {
		if i&1 != 0 {
			pow *= sq
		}
		sq *= sq
	}
	return hash, pow
}

func indexRabinKarp(s, substr string) int {
	hashss, pow := hashStr(substr) // 待匹配的hash值 以及primeRK^对应长度
	n := len(substr)
	var h uint32

	// 计算目标字符串前n位hash并与匹配字符串hash进行对比 hash相同就返回下标
	for i := 0; i < n; i++ {
		h = h*primeRK + uint32(s[i])
	}
	if h == hashss && s[:n] == substr {
		return 0
	}

	// 如果前面不相同就开始不断匹配后面的
	for i := n; i < len(s); {
		h *= primeRK
		h += uint32(s[i])
		h -= pow * uint32(s[i-n])
		i++
		if h == hashss && s[i-n:i] == substr {
			return i - n
		}
	}
	return -1
}

// Rabin-Karp匹配
func RabinKarp(haystack, needle string) int {
	n, m := len(haystack), len(needle)
	if m == 0 {
		return 0
	}

	var k1 int = 1000000000 + 7
	var k2 int = 1337
	rand.Seed(time.Now().Unix())
	var kMod1 int64 = int64(rand.Intn(k1)) + int64(k1)
	var kMod2 int64 = int64(rand.Intn(k2)) + int64(k2)

	var hash_needle int64 = 0
	for i := 0; i < m; i++ {
		hash_needle = (hash_needle*kMod2 + int64(needle[i])) % kMod1
	}
	var hash_haystack int64 = 0
	var extra int64 = 1
	for i := 0; i < m-1; i++ {
		hash_haystack = (hash_haystack*kMod2 + int64(haystack[i%n])) % kMod1
		extra = (extra * kMod2) % kMod1
	}
	for i := m - 1; (i - m + 1) < n; i++ {
		hash_haystack = (hash_haystack*kMod2 + int64(haystack[i%n])) % kMod1
		if hash_haystack == hash_needle {
			return i - m + 1
		}
		hash_haystack = (hash_haystack - extra*int64(haystack[(i-m+1)%n])) % kMod1
		hash_haystack = (hash_haystack + kMod1) % kMod1
	}
	return -1
}
```

## sunday

在匹配失败时关注文本串中参加匹配的最末位字符的下一位字符（暴力匹配匹配串在失败的时候每次只能移动一位），根据haystack参与的末尾下一位判断在不在匹配串中，在的话移动指定位置对齐然后进行判断，如果不在的话直接跳过移动整个匹配串的长度。

```Go
func SundaySearch(haystack string, needle string) int {
	if len(needle) == 0 {
		return 0
	}

	if len(haystack) == 0 && len(needle) != 0 {
		return -1
	}

	keys := make([]int, 256) // 记录ascii对应字符在字符串最右边的位置
	for i := 0; i < 256; i++ {
		keys[i] = -1
	}
	for i := 0; i < len(needle); i++ {
		keys[needle[i]] = i // 记录每个字符在字符串哪个位置，会被后面的覆盖 记录最右侧位置
	}

	//
	sIndex, pIndex, space := 0, 0, 0 // 双指针
	for sIndex < len(haystack) {
		if haystack[sIndex] == needle[pIndex] {
			sIndex++
			pIndex++
			if pIndex == len(needle) {
				return sIndex - len(needle)
			}
		} else {
			pIndex = 0
			if space+len(needle) < len(haystack) {
				space += len(needle) - keys[haystack[space+len(needle)]] // !寻找下个模式的匹配初始位置
			} else {
				return -1
			}
			sIndex = space
		}
	}
	return -1
}
```

## KMP

>时间复杂度： O(m+n) ；空间复杂度在；O(m)

利用已经部分匹配这个有效信息，保持主串上的指针不回溯，通过修改子串的指针，让模式串尽量地移动到有效的位置。

创建模式串的next（每次遍历失败的部分查看前后缀是否相同，相同的话下一个匹配位置就是相同的下一个），表示当两个字符串进行模式匹配失败的时候，需要从模式串的哪一个位置重新开始匹配，利用这next数组，来减少字符比较的次数，当比较字符失败时，就根据next换到新的位置。

```Go
func StringKMP(mainStr, subStr string) int {
    if len(mainStr) < len(subStr) {
        return -1
    }
    
    // 构建next数组 ！！！注意这里在后续匹配失败时不是直接移动到指定位置，而是用来标识有哪些部分确实是有序的，使得能够直接跳过已知能够匹配的部
    nextArr := make([]int, len(subStr))
    for i := 0; i < len(subStr); i++ {
        t := -1
        for j := 0; j < i-1; j++ {
            if subStr[:j+1] == subStr[i-(j+1):i] {
                t = j + 1 // 有相同的前后缀则将其移动到j后面一个
            }
        }
        nextArr[i] = t // 表示在i前面与[i,...]相同的前缀[...,j-1]
    }
    
    // kmp
    i, y, index := 0, 0, []int{}
    // i+y: mainStr开始匹配的位置 y: subStr开始匹配的位置 index: 结果
    for i <= len(mainStr) - len(subStr) {
        for j := y; j <= len(subStr); j++ {
            // 匹配完成 可以继续匹配下一个 也可以直接返回
            if j == len(subStr) {
                index = append(index, i)
                i++
                y=0
                break
            }
            
            // 相等则继续
            if mainStr[i+j] == subStr[j] {
                continue
            }
            
            // 匹配失败 根据nextArr快速查找下一个
            idx := j
            for ; idx > 1; idx-- {
                if nextArr[idx] >= 1 {
                    i = i + (j - nextArr[j])
                    y = nextArr[j]
                    break
                }                
            }
            if idx <= 1 {
                i++
                y=0
            }
            break
        }
    }
    return inde
}
```

```Go
type str struct {
	i   int //位置
	num int //数值
}

func Manacher(s string) string {
	length := len(s)
	btyestr := []byte(s)

	if length < 2 {
		return s
	}

	bytestr2 := make([]byte, 2*length+1)
	p := make([]int, 2*length+1)

	for m := 0; m < length; m++ {

		bytestr2[2*m] = 1
		bytestr2[2*m+1] = btyestr[m]

	}
	bytestr2[2*length] = 1

	fmt.Println(bytestr2)

	//在每个字符间插入相同的字符
	for i := 0; i < len(bytestr2); i++ {

		p[i] = 1
		if i != 0 && i != 2*length {
			for j := 1; ; j++ {
				if bytestr2[i-j] == bytestr2[i+j] {
					p[i] = p[i] + 1
				} else {
					break
				}
				if i-j == 0 || i+j == 2*length {
					break
				}
			}
		}
	}

	maxdata := str{0, 1}

	for j := 0; j < 2*length+1; j++ {

		if p[j] > maxdata.num {

			maxdata.i = j
			maxdata.num = p[j]
		}
	}

	bytestr3 := make([]byte, maxdata.num-1)
	//找出最长子串，跳过之前我自己添加的字符
	for h := 0; h < maxdata.num-1; h++ {
		if (maxdata.num-1)%2 == 0 {
			//跳过之前我自己添加的字符，所以+2*h
			bytestr3[h] = bytestr2[maxdata.i+2-maxdata.num+2*h]
		} else {
			bytestr3[h] = bytestr2[maxdata.i+2-maxdata.num+2*h]
		}

	}

	ss := string(bytestr3)
	return ss
}
```



## 寻找最近的回文数

给定一个数字字符串，寻找离它最近的回文数(不包括自身，如果有多个值，返回比较小的一个)

```go
func nearestPalindromic(n string) string {
	m := len(n)
	candidates := []int{int(math.Pow10(m-1))-1, int(math.Pow10(m)) + 1} // 9..9 10..01
	
	// 自己前半部分
	selfPrefix, _ := strconv.Atoi(n[:(m+1)/2])
	for _, x := range []int{selfPrefix-1, selfPrefix, selfPrefix+1} {
		y := x
		if m & 1 == 1 {
			y /= 10 // 奇数个数
		}
		for ; y > 0; y/=10 {
			x = x*10 + y%10
		}
		candidates = append(candidates, x)
	}
	
	ans := -1
	selfNumber, _ := strconv.Atoi(n)
	for _, candidate := range candidates {
		if candidate != selfNumber {
			if ans == -1 ||
			abs(candidate-selfNumber) < (ans-selfNumber) ||
			abs(candidate-selfNumber) == abs(ans-selfNumber)&&candidate<ans {
				ans = candidate
			}
		}
	}
	return strconv.Itoa(ans)
}
```

## 统计删除子串能获得的回文串个数
定义`f(x, i, j)`为字符串`[i, j]`区间以x为首字母的能构成回文串的个数

设字符集种类为n，`x[k]`为第k种字符

- 当`s[i]=x, s[j]x`： `f(x, i, j) = ∑(k=0,n)f(x[k], i+1, j-1) + 2`
- 当`s[i]=x，s[j]≠x`：`f(x, i, j) = f(x, i, j-1)`
- 当`s[i]≠x，s[j]=x`：`f(x, i, j) = f(x, i+1, j)`
- 当`s[i]≠x, s[j]≠x`：`f(x, i, j) = f(x, i+1, j-1)`

```go
// 这里假设仅包含a b c d
func countPalindromicSubsequences(s string) (ans int) {
	const mod int = 1e9 + 7
	n := len(s)
	dp := [4][][]int{}
	for i := range dp {
		dp[i] = make([][]int, n)
		for j := range dp[i] {
			dp[i][j] = make([]int, n)
		}
	}
	// dp初始化
	for i, c := range s {
		dp[c-'a'][i][i] = 1
	}
	// 状态转移
	for sz := 2; sz <= n; sz++ {
		// 状态的基础数据是中间的，sz是滑动窗口的大小
		for i, j := 0, sz-1; j < n; i, j= i+1, j+1 {
			for k, c := 0, byte('a'); k < 4; k++ {
				// 枚举a b c d
				if s[i] == c && s[j] == c {
					dp[k][i][j] = (2 + dp[0][i+1][j-1] + dp[1][i+1][j-1] + dp[2][i+1][j-1] + dp[3][i+1][j-1]) % mod
				} else if s[i] == c && s[j] != c {
					dp[k][i][j] = dp[k][i][j-1]
				} else if s[i] != c && s[j] == c {
					dp[k][i][j] = dp[k][i+1][j]
				} else {
					dp[k][i][j] = dp[k][i+1][j-1]
				}
				c++
			}
		}
	}
	
	for _, d := range dp {
		ans += d[0][n-1]
	}
	return ans % mod
}
```

## 最大的回文乘积

给定n，判断两个n位数所乘的最大回文值

#枚举

两个n位数相乘最多`2*n`, 枚举前半部分2^10, 然后翻转判断能否被两个整数相乘得到

```go
func largestPalindrome(n int) int {
	if n == 1 {
		return 9
	}
	upper := int(math.Pow10(n)) - 1
	for left := upper; ; left-- {
		p := left
		for x := left; x > 0; x /= 10 {
			p = p*10 + x%10
		}
		for x := upper; x*x >= p; x-- {
			// 只取x*x>=p，因为小的数在之前取大的数的时候已经计算过了
			if p % x == 0 {
				return p % 1337
			}
		}
	}
}
```

## 最长特殊子序列

给定一个字符串数组，需要找到最长的子序列(可以不连续)字符串，满足不在其他字符串中以子序列的形式出现

`基本原则`

特殊子序列表示不在其他字符串中以子序列存在

`推论`

如果一个子序列是特殊子序列，那么继续添加新的字符，依然会是特殊子序列

所以这个题就转变成了寻找一个字符串不在其他字符串中以子序列存在，使用双指针判断一个字符串是否是另一个的子序列

那岂不是字符串中最长的就是最长特殊子序列，`["aaa", "aaa", "aa"]`, 不过这种为-1，就是最长第字符串中，必须有个数为1个的例子

> 这里陷入了误区，不要老是盯着最长的字符串，还有第二长、第三长的也可能满足呢

#双指针

```go
func findLUSlength(strs []string) int {
	res := -1
next:
	for i, base := range strs {
		for j, target := range strs {
			if i != j && isSub(base, target) {
				continue next
			}
		}
		if len(base) > res {
			res = len(base)
		}
	}
	return res
}

func isSub(a, b string) bool {
	if len(a) > len(b) {
		return false
	}
	
	i, j := 0, 0
	aLen, bLen := len(a), len(b)
	for j < bLen {
		if a[i] == b[j] {
			if i++; i == aLen {
				return true
			}
			j++
		} else {
			j++
		}
	}
	return false
}
```

## 求解同时出现大小写的子字符串

给定一个字符串，例如`YazaAay`，求解子字符串，满足其中相应的大小写同时存在

#枚举

使用状态压缩进行优化

由于大小写字母分别只有26个，使用`int`类型表示当前存在哪些符号

在枚举所有的`i...j`时不断更新upper、lower，当两个值相同时就是满足条件的子字符串，然后更新最大值


```go
func longestNiceSubstring(s string) {
	ans := ""
	for i := range s {
		lower, upper := 0, 0
		for j := 1; j < len(s); j++ {
			if unicode.IsLower(rune(s[j])) {
				lower |= 1 << (s[j] - 'a')
			} else {
				upper |= 1 << (s[j] - 'A')
			}
			if lower == upper && j-i+1 > len(ans) {
				ans = s[i:j+1]
			}
		}
	}
	return ans
}
```

```go
func deserialize(s string) *NestedInteger {
	if s[0] != '[' {
		num, _ := strconv.Atoi(s)
		ni := &NestedInteger{}
		ni.SetInteger(num)
		return ni
	}
	
	stack, num, negative := []*NestedInteger{}, 0, false
	for i, ch := range s {
		if ch == '-' {
			negative = true
		} else if unicode.IsDigit(ch) {
			num = num*10 + int(ch-'0')
		} else if ch == '[' {
			stack = append(stack, &NestedInteger{})
		} else if ch == ',' || ch == ']' {
			if unicode.IsDigit(rune(s[i-1])) {
				if negative {
					num = -num
				}
				ni := NestedInteger{}
				ni.SetInteger(num)
				stack[len(stack)-1].Add(ni)
			}
			
		}
		num, negative = 0, false
		if ch == ']' && len(stack) > 1 {
			stack[len(stack)-2].Add(*stack[len(stack)-1])
			stack = stack[:len(stack)-1]
		}
	}
	return stack[len(stack)-1]
}
```
