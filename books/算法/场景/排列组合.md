返回括号所有可能的组成方式

设`dp[i]`表示长度为i的，可能的组合方式

```go
func generateParenthesis(n int) []string {
    if n == 0 {
        return []string{}
    }

    dp := make([][]string, n+1)
    for i := range dp {
        dp[i] = []string{}
    }
    dp[0] = []string{""}
    dp[1] = []string{"()"}
    for i := 2; i <= n; i++ {
        cur := []string{}
        for j := 0; j < i; j++ {
            q := i - j - 1
            for _, itema := range dp[j] {
                for _, itemb := range dp[q] {
                    cur = append(cur, fmt.Sprintf("(%s)%s", itema, itemb))
                }
            }
        }
        dp[i] = cur
    }
    return dp[n]
}
```
