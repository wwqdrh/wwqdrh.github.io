# 寻找有多少种路径的可能

## `m*n`棋盘上有障碍物有多少种路径的可能

```go
func uniquePathsWithObstacles(obstacleGrid [][]int) int {
    // 设dp[i][j]表示从i,j出发到达终点的路径总和
    if len(obstacleGrid) == 0 {
        return 0
    }
    m, n := len(obstacleGrid), len(obstacleGrid[0])
    dp := make([]int, n) // 滚动数组的思想，优化空间复杂度
    if obstacleGrid[m-1][n-1] == 1 {
        return 0
    }

    dp[n-1] = 1 // 终点没有障碍物的时候
    for i := m - 1; i >= 0; i-- {
        for j := n - 1; j >= 0; j-- {
            if obstacleGrid[i][j] == 1 {
                dp[j] = 0
                continue
            }
            if j < n - 1 {
                dp[j] += dp[j+1]
            }
        }
    }
    return dp[0]
}
```
