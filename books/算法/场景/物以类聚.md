并查集， 树形数据结构，特点是可以通过子节点找到父节点，用于处理一些不交集的合并及查询问题

判断是否是一类的就是通过最顶级的位置上进行判断的，rank记录合并次数，合并次数越多越能成为主节点

> 下面的代码可能会根据实际情况稍微有一点变种

```go
// 模板代码
type unionFind struct {
	parent, rank []int
}

func NewUnionFind(n int) *unionFind {
	parent := make([]int, n)
	for i := range parent {
		parent[i] = i // 初始状态每个节点的父节点是自己
	}
	return &unionFind{parent, make([]int, n)}
}

func (uf *unionFind) find(x int) int { 
	if uf.parent[x] != x {
		uf.parent[x] = uf.find(uf.parent[x])
	}
	return uf.parent[x]
}

func (uf *unionFind) merge(x, y int) {
	x, y = uf.find(x), uf.find(y)
	if x == y {
		return // 已经是一个分类了
	}
	if uf.rank[x] > uf.rank[y] {
		uf.parent[y] = x
	} else if uf.rank[x] < uf.rank[y] {
		uf.parent[x] = y
	} else {
		uf.parent[y] = x
		uf.rank[x]++ // x的合并+1
	}
}
```

相似字符串分组

给定的字符串是异位次，相似的字符串是相同位置上不同字符的个数只有两个

并查集模板题

```go
type unionFind struct {
	parent, rank []int
	count int // 连通分量数
}

func NewUnionFind(n int) *unionFind {
	parent := make([]int, n)
	rank := make([]int, n)
	for i := range parent {
		parent[i] = i
		rank[i] = 1
	}
	return &unionFind{
		parent: parent,
		rank: rank,
		count: n,
	}
}

func (uf *unionFind) find(x int) int {
	if uf.parent[x] != x {
		uf.parent[x] = uf.find(uf.parent[x])
	}
	return uf.parent[x]
}

func (uf *unionFind) merge(x, y int) {
	fx, fy := uf.find(x), uf.find(y)
	if fx == fy {
		return
	}
	if uf.rank[fx] < uf.rank[fy] {
		fx, fy = fy, fx
	}
	uf.parent[fy] = fx
	uf.rank[fx] += uf.rank[fy]
	uf.count--
}

func numSimilarGroups(strs []string) int {
	n := len(strs)
	uf := NewUnionFind(n)
	for i, s := range strs {
		for j := i+1; j < n; j++ {
			if uf.find(i) != uf.find(j) && isSimilar(s, strs[j]) {
				uf.merge(i, j)
			}
		}
	}
	return uf.count
}

func isSimilar(a, b string) bool {
	if len(a) != len(b) {
		return false
	}
	diff := 0
	for i := range a {
		if a[i] != b[i] {
			diff++
			if diff > 2 {
				return false
			}
		}
	}
	return true
}
```
