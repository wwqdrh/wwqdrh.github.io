给定四个坐标判断是否是一个合法的正方形

每次选择三个点判断是否是直角等腰三角形，方法就是分别求出三条边的长度的平方(`(x1-x2)^2 + (y1-y2)^2`)

判断是否能满足`l1+l2=l3`, 或者其他等式，并且更新边的长度

```go
func validSquare(p1 []int, p2 []int, p3 []int, p4 []int) bool {
    l1 := checkAndLength(p1, p2, p3)
    if l1 == -1 {
        return false
    }
    l2 := checkAndLength(p1, p2, p4)
    if l2 == -1 {
        return false
    }
    l3 := checkAndLength(p2, p3, p4)
    if l3 == -1 {
        return false
    }
    return l1 == l2 && l2 == l3
}

func checkAndLength(p1 []int, p2 []int, p3 []int) int {
    l1 := (p1[0] - p2[0]) * (p1[0] - p2[0]) + (p1[1] - p2[1]) * (p1[1] - p2[1])
    l2 := (p1[0] - p3[0]) * (p1[0] - p3[0]) + (p1[1] - p3[1]) * (p1[1] - p3[1])
    l3 := (p2[0] - p3[0]) * (p2[0] - p3[0]) + (p2[1] - p3[1]) * (p2[1] - p3[1])
    if l1 + l2 != l3 && l1 + l3 != l2 && l2 + l3 != l1 {
        return -1
    }

    if l1 == 0 || l2 == 0 {
        return -1 // 不能共点
    }

    if l1 < l2 {
        return l1
    } else {
        return l2
    }
}
```
