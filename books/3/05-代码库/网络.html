套接字

用于进程之间的通信方式

服务端

package main

import "net"

func main() {
    l, err := net.Listen("unix", "/tmp/echo.sock")
    if err != nil {
        println("listen error", err.Error())
        return
    }

    for {
        fd, err := l.Accept()
        if err != nil {
            println("accept error", err.Error())
            return
        }

        go echoServer(fd)
    }
}


func echoServer(c net.Conn) {
    for {
        buf := make([]byte, 512)
        nr, err := c.Read(buf)
        if err != nil {
            return
        }

        data := buf[0:nr]
        println("Server got:", string(data))
        _, err = c.Write(data)
        if err != nil {
            panic("Write: " + err.Error())
        }
    }
}

客户端

package main

import (
    "io"
    "net"
    "time"
)

func main() {
    c, err := net.Dial("unix", "/tmp/echo.sock")
    if err != nil {
        panic(err.Error())
    }
    defer c.Close()

    go reader(c)
    for {
        _, err := c.Write([]byte("hi...."))
        if err != nil {
            println(err.Error())
            break
        }
        time.Sleep(1e9)
    }
}

func reader(r io.Reader) {
    buf := make([]byte, 1024)
    for {
        n, err := r.Read(buf[:])
        if err != nil {
            return
        }
        println("Client got:", string(buf[0:n]))
    }
}
