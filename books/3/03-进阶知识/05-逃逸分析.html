在编译原理中，分析指针动态范围的方法称之为逃逸分析。通俗来讲，当一个对象的指针被多个方法或线程引用时，我们称这个指针发生了逃逸。

go中就是用来判断是放在栈上还是放在堆上

  • 需要被外部使用的就会在堆上

通过golang自带的trace工具进行分析

下面是一个示例

import (
	"fmt"
	"os"
	"runtime"
	"runtime/trace"
	"time"
)

func numFinizaler() {
	var i = 3
	// 这里的func 就是后面一直说的 finalizer
	runtime.SetFinalizer(&i, func(i *int) {
		fmt.Println(i, *i, "set finalizer")
	})
}

func main() {
	f, _ := os.Create("trace.out")
	defer f.Close()
	trace.Start(f)
	defer trace.Stop()

	numFinizaler()
	runtime.GC()
	time.Sleep(3 * time.Second)
	runtime.GC()
}

go run ..
