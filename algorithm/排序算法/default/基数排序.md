func max(nums ...int) int {
	max := 0
	for _, num := range nums {
		if num > max {
			max = num
		}
	}
	return max
}

func pow10(num int) int {
    res := 1
    base := 10
    for num > 0 {
        if num & 1 == 1 {
            num--
            res *= base
        }
        num >>= 1
        base *= base
    }
    return res
}

func radixSort(nums []int) []int {
    numberBit := len(fmt.Sprint(max(nums)))
    for i := 0; i < numberBit; i++ {
		rec := make([][]int, 10)

		for _, num := range nums {
            c := (num/pow10(i))%10
			rec[c] = append(rec[c], num)
		}
		numsCopy := make([]int, 0)
		for j := 0; j < 10; j++ {
			numsCopy = append(numsCopy, rec[j]...)
		}
		nums = numsCopy
	}
	return nums
}
