func bubbleSort(nums []int) []int {
    nums := nums[:]
    length := len(nums)
    for i := 0; i < length; i++ {
        for j := 0; j < length-i-1; j++ {
            if nums[j] > nums[j+1] {
                nums[j], nums[j+1] = nums[j+1], nums[j]
            }
        }
    }
    return nums
}
