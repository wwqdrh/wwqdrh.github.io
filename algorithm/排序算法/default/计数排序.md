func max(nums ...int) int {
    maxVal := -1 << 31
    for _, item := range nums {
        if item > maxVal {
            maxVal = item
        }
    }
    return maxVal
}

// 计数排序
func countSort(nums []int) []int {
    maxVal := max(nums...)
    bucket := make([]int, maxVal+1)
    for _, num := range nums {
        bucket[num]++
    }
    
    res := []int{}
    for i, cnt := range bucket {
        for ; cnt > 0; cnt-- {
            res = append(res, i)
        }
    }
    return res
}
