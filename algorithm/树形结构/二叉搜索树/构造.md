给定一个数组创建一个二叉搜索树

二叉搜索树的创建分为不平衡，平衡，随机，向左倾斜，向右倾斜几种

```go
//定义二叉树
type BinareSearchTree struct {    
	Root *TreeNode
}

//定义二叉树节点
type TreeNode struct {    
	Val int    
	Left *TreeNode    
	Right *TreeNode
}

func NewBST(nums []int) *BinareSearchTree {
    root := &TreeNode{nums[0],nil,nil}//需要先初始化一个根节点
    for _, num := range nums[1:] {
        root = insertIntoBST(root, num)
    }
    return &{BinareSearchTree: root}
}

func insertIntoBST(root *TreeNode, val int) *TreeNode {
    if root == nil {
        return &TreeNode{Val: val}
    }
    p := root
    for p != nil {
        if val < p.Val {
            if p.Left == nil {
                p.Left = &TreeNode{Val: val}
                break
            }
            p = p.Left
        } else {
            if p.Right == nil {
                p.Right = &TreeNode{Val: val}
                break
            }
            p = p.Right
        }
    }
    return root
}
```
