假设图中的一条边的两个端点可以分成两个不同的子集就满足二分图

染色问题中出现，即相邻的两个要不是同一种类型

还有就是电影和演员，如果用哈希表存储，需要两个哈希表分别存储「每个演员到电影列表」的映射和「每部电影到演员列表」的映射。但如果用「图」结构存储，将电影和参演的演员连接，很自然地就成为了一幅二分图

```go
func possibleBipartition(n int, dislikes [][]int) bool {
    // 染色法
    graph := make([][]int, n)
    for _, item := range dislikes {
        i, j := item[0]-1, item[1]-1
        graph[i] = append(graph[i], j)
        graph[j] = append(graph[j], i)
    }

    colors := make([]int, n)
    var dfs func(node int, color int) bool
    dfs = func(node int, color int) bool {
        colors[node] = color
        for _, next := range graph[node] {
            if colors[next] == color || colors[next] == 0 && !dfs(next, color^3) {
                return false
            }
        }
        return true
    }

    for i := 0; i < n; i++ {
        if colors[i] == 0 {
            if !dfs(i, 1) {
                return false
            }
        }
    }
    return true
}
```
