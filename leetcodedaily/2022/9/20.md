dfs

使用一个int表示集合中选择的状态，有哪些还可以选

```go
func canPartitionKSubsets(nums []int, k int) bool {
    totalNums := 0
    for _, item := range nums {
        totalNums += item
    }
    if totalNums % k != 0 {
        return false
    }
    sort.Ints(nums)
    n := len(nums)
    per := totalNums / k
    if nums[n-1] > per {
        return false
    }

    // mem
    mem := make([]bool, 1<<n)
    for i := range mem {
        mem[i] = true
    }
    var dfs func(state int, cur int) bool
    dfs = func(state, cur int) bool {
        if state == 0 {
            return true
        }
        if !mem[state] {
            return false
        }
        mem[state] = false
        for i, num := range nums {
            if cur + num > per {
                break
            }
            if state & (1 << i) > 0 && dfs(state ^ (1 << i), (cur + num)%per) {
                return true
            }
        }
        return false
    }
    return dfs((1<<n)-1, 0)
}
```
