简单模拟

记录可以作为开始的区块以及区块的对应

如果找不到开始区块就不能构成

```go
func canFormArray(arr []int, pieces [][]int) bool {
    piecemap := map[int][]int{}
    for _, pice := range pieces {
        piecemap[pice[0]] = pice
    }

    for i, n := 0, len(arr); i < n; {
        cur, ok := piecemap[arr[i]]
        if !ok {
            return false
        }
        j := 0
        for ;j < len(cur);j++ {
            if arr[i+j] != cur[j] {
                break
            }
        }
        if j < len(cur) {
            return false
        }
        i += j
    }
    return true
}
```
