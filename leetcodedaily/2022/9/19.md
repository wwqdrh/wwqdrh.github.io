频率计数，桶，下标是频率，每个元素是一个set

相同桶里面的按照数值大小进行比较

```go
func frequencySort(nums []int) []int {
    bucket := make([]map[int]struct{}, len(nums)+1)
    for i := range bucket {
        bucket[i] = map[int]struct{}{}
    }
    freq := map[int]int{}

    for _, num := range nums {
        freq[num]++
        delete(bucket[freq[num]-1], num)
        bucket[freq[num]][num] = struct{}{}
    }

    res := []int{}
    for _, buck := range bucket {
        curs := []int{}
        for val := range buck {
            curs = append(curs, val)
        }
        sort.Slice(curs, func(i, j int)bool{return curs[i] > curs[j]})
        for _, val := range curs {
            for i := 0; i < freq[val]; i++ {
                res = append(res, val)
            }
        }
    }
    return res
}
```
