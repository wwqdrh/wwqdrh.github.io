func minOperations(logs []string) int {
    stack := []string{}
    for _, item := range logs {
        switch item {
        case "./":
        case "../":
            if len(stack) > 0 {
                stack = stack[:len(stack)-1]
            }
        default:
            stack = append(stack, item)
        }
    }
    return len(stack)
}
