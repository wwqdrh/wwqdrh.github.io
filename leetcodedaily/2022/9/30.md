遍历两次

一次标记

一次清除

```go
func setZeroes(matrix [][]int)  {
    zeror, zeroc := map[int]struct{}{}, map[int]struct{}{}
    for i, row := range matrix {
        for j, item := range row {
            if item == 0 {
                zeror[i] = struct{}{}
                zeroc[j] = struct{}{}
            }
        }
    }

    for i, row := range matrix {
        for j := range row {
            if _, ok := zeror[i]; ok {
                matrix[i][j] = 0
            } else if _, ok := zeroc[j]; ok {
                matrix[i][j] = 0
            }
        }
    }
}
```
