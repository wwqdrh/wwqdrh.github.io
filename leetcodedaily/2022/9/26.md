位运算的应用

异或

既然一个序列要找两个缺失的找不出来那么就把它分成两个序列
分组规则当然不能是奇偶这种规则，因为可能缺失的同时为奇
按照全部1...n与nums全部异或后的位来判断，某一位是1，那么其中一个数在这一位是一，另外一个数在这一位是0

```go
func missingTwo(nums []int) []int {
    xornum := 0
    n := len(nums) + 2
    for _, num := range nums {
        xornum ^= num
    }  
    for i := 1; i <= n; i++ {
        xornum ^= i
    }

    lowbit := xornum & -xornum
    num1, num2 := 0, 0
    for _, num := range nums {
        if num & lowbit > 0 {
            num1 ^= num
        } else {
            num2 ^= num
        }
    }
    for i := 1; i <= n; i++ {
        if i & lowbit > 0 {
            num1 ^= i
        } else {
            num2 ^= i
        }
    }
    return []int{num1, num2}
}
```
