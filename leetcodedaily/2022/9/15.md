```go
func flipLights(n int, presses int) int {
    // 虽然看着长度不大，好像可以用dfs，但是实际复杂度在 4^1000 这是很高的
    // 即使有记忆化搜索 但是会有超出内存限制的问题

    // 需要压缩状态，这里是压缩长度和只考虑开关操作的奇偶
    // 找规律，四个开关使得灯泡变换，最小公倍数为6，所以以6为周期
    if n >= 6 {
        n = 6
    }
    opts := []int{0b111111, 0b010101, 0b101010, 0b001001}
    res := map[int]struct{}{}
    for mask, state := 0, 1 << 4; mask < state; mask++ {
        // state 位 0表示偶数次 1表示奇数次
        cur := bits.OnesCount(uint(mask))
        if cur <= presses && cur & 1 == presses & 1 {
            // 比peesses小并且奇偶相同
            targetState := 0
            for i, opt := range opts {
                if (mask >> i) & 1 == 1 {
                    // 说明这个按键按了一次
                    targetState ^= opt
                }
            }

            k := 6
            if n < k {
                k = n
            }
            targetState &= (1 << k) - 1
            // targetState >>= (6 - n)
            res[targetState] = struct{}{}
        }
    }
    return len(res)
}
```
