将每行和每列的个数记录下

```go
func numSpecial(mat [][]int) int {
    if len(mat) == 0 {
        return 0
    }
    
    rowCnt, colCnt := make([]int, len(mat)), make([]int, len(mat[0]))
    for i, line := range mat {
        for j, item := range line {
            if item == 1 {
                rowCnt[i]++
                colCnt[j]++
            }
        }
    }

    res := 0
    for i, line := range mat {
        for j, item := range line {
            if item ==1 && rowCnt[i] - item == 0 && colCnt[j] - item == 0 {
                res++
            }
        }
    }
    return res
}
```
