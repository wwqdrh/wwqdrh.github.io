func trimMean(arr []int) float64 {
    n := len(arr)
    left, right := n / 20, 19*n/20
    sort.Slice(arr, func(i, j int) bool {
        return arr[i] < arr[j]
    })

    sum := 0
    for _, item := range arr[left:right] {
        sum += item
    }
    return float64(sum) / float64(right - left)
}
