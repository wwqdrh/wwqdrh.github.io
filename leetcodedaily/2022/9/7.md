func reorderSpaces(text string) string {
    words := strings.Fields(text)
    spaces := strings.Count(text, " ")
    lw := len(words) - 1
    if lw == 0 {
        return words[0] + strings.Repeat(" ", spaces)
    }
    return strings.Join(words, strings.Repeat(" ", spaces / lw)) + strings.Repeat(" ", spaces % lw)
}
