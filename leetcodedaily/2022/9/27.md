```go
func CheckPermutation(s1 string, s2 string) bool {
    cnt := map[rune]int{}
    for _, ch := range s1 {
        cnt[ch]++
    }
    for _, ch := range s2 {
        cnt[ch]--
    }
    for _, val := range cnt {
        if val != 0 {
            return false
        }
    }
    return true
}
```
