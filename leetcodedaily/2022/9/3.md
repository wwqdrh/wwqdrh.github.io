func findLongestChain(pairs [][]int) int {
    sort.Slice(pairs, func(i, j int) bool {
        return pairs[i][0] <= pairs[j][0] 
    })

    // 设dp[i]为以第i为结尾最长的数对链
    res := 0
    n := len(pairs)
    dp := make([]int, n)
    for i := range pairs {
        dp[i] = 1 // 初始值在这设置免得多遍历一次
        for j := range pairs[:i] {
            if pairs[j][1] < pairs[i][0] {
                dp[i] = max(dp[i], dp[j]+1)
            }
        }
        res = max(res, dp[i])
    }
    return res
}

func max(a, b int) int {
    if a >= b {
        return a
    }
    return b
}
