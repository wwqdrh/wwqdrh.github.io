简单模拟

由于k限制了长度

所以可以简单的将数据扩大两倍避免取模运算

```go
func decrypt(code []int, k int) []int {
    n := len(code)
    res := make([]int, n)
    
    code = append(code, code...) // 避免取模
    l, r := 1, k
    if k < 0 {
        l, r = n + k, n - 1
    }
    cur := 0
    for i := l; i <= r; i++ {
        cur += code[i]
    }
    for i := range res {
        res[i] = cur
        cur -= code[l]
        cur += code[r+1]
        l, r = l+1, r+1
    }
    return res
}
```
