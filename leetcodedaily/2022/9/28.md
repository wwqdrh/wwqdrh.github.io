func getKthMagicNumber(k int) int {
    dp := make([]int, k)
    dp[0] = 1
    id1, id2, id3 := 0, 0, 0 // 3 5 7
    for i := 1; i < k; i++ {
        cur := min(dp[id1]*3, dp[id2]*5, dp[id3]*7)
        dp[i] = cur
        if cur == dp[id1]*3 {
            id1++
        }
        if cur == dp[id2]*5 {
            id2++
        }
        if cur == dp[id3]*7 {
            id3++
        }
    }
    return dp[k-1]
}

func min(num int, nums ...int) int {
    for _, item := range nums {
        if item < num {
            num = item
        }
    }
    return num
}
