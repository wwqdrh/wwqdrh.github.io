扫描线模板题

```go
const MOD = int64(1e9 + 7)

func rectangleArea(rectangles [][]int) int {
    list := []int{}
    // 将x先加入并排序
    for _, info := range rectangles {
        list = append(list, info[0], info[2])
    }
    sort.Ints(list)
    ans := int64(0)
    for i := 1; i < len(list); i++ {
        a, b := list[i-1], list[i]
        length := b - a
        if length == 0 {
            continue
        }

        // 将该x区域包裹在内的y线加入到队列
        lines := [][]int{}
        for _, info := range rectangles {
            if info[0] <= a && b <= info[2] {
                lines = append(lines, []int{info[1], info[3]})
            }
        }
        sort.Slice(lines, func(i, j int) bool {
            if lines[i][0] != lines[j][0] {
                return lines[i][0] - lines[j][0] < 0
            }
            return lines[i][1] - lines[j][1] < 0
        })
        total, l, r := int64(0), -1, -1
        for _, cur := range lines {
            if cur[0] > r {
                total += int64(r - l)
                l, r = cur[0], cur[1]
            } else if cur[1] > r {
                r = cur[1]
            }
        }
        total += int64(r - l)
        ans += total * int64(length)
        ans %= MOD
    }
    return int(ans)
}
```
