```go
func finalPrices(prices []int) []int {
    res := []int{}

    n := len(prices)
    for i, price := range prices {
        diff := 0
        for j := i + 1; j < n; j++ {
            if prices[j] <= prices[i] {
                diff = prices[j]
                break
            }
        }
        res = append(res, price - diff)
    }
    return res
}
```
