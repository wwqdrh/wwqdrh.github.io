简单遍历

存储元素出现的最左边的下标

```go
func maxLengthBetweenEqualCharacters(s string) int {
    prevId := map[rune]int{}
    res := -1
    for i, ch := range s {
        if pre, ok := prevId[ch]; ok {
            if i - pre - 1 > res {
                res = i - pre - 1
            }
        } else {
            prevId[ch] = i
        }
    }
    return res
}
```
