多块岛屿

将每块岛屿散列化成左上角的元素

然后遍历所有为0位置的元素，计算最大的联通量

```go
func largestIsland(grid [][]int) int {
    n := len(grid)
    res := 0
    // 将n个独立的连接的区域进行离散化各自独立成一块
    area := map[int]int{}
    tag := map[int]int{}
    dires := [][]int{
        {0,1},{0,-1},{1,0},{-1,0},
    }
    var dfs func(i, j, t int)
    dfs = func(i, j, t int) {
        tag[i*n+j+1] = t
        area[t]++
        for _, dir := range dires {
            ni, nj := i+dir[0], j+dir[1]
            if ni >= 0 && ni < n && nj >= 0 && nj < n && grid[ni][nj] == 1 && tag[ni*n+nj+1] == 0 {
                dfs(ni, nj, t)
            }
        }
    }

    for i, line := range grid {
        for j, item := range line {
            if item == 1 {
                if _, ok := tag[i*n+j+1]; !ok {
                    curTag := i * n + j+1
                    dfs(i, j, curTag)
                    if area[curTag] > res {
                        res = area[curTag]
                    }
                }
            }
        }
    }

    // 从0的地方开始连接
    for i, row := range grid {
        for j, item := range row {
            if item == 0 {
                total := 1
                added := map[int]struct{}{}
                for _, dire := range dires {
                    ni, nj := i+dire[0], j+dire[1]
                    t := tag[ni * n + nj+1]
                    if ni >= 0 && ni < n && nj >= 0 && nj < n {
                        if _, ok := added[t]; !ok {
                            total += area[t]
                            added[t] = struct{}{}
                        }
                    }

                }
                if total > res {
                    res = total
                }
            }
            
        }
    }
    return res
}
```
