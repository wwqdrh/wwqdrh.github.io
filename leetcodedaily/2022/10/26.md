func shortestSubarray(nums []int, k int) int {
    prefix := make([]int, len(nums)+1)
    for i, num := range nums {
        prefix[i+1] = prefix[i] + num
    }

    res := int(1e9)
    queue := []int{}
    for i, pre := range prefix {
        for len(queue) > 0 && pre < prefix[queue[len(queue)-1]] {
            queue = queue[:len(queue)-1]
        }
        for len(queue) > 0 && pre - prefix[queue[0]] >= k {
            idx := queue[0]
            queue = queue[1:]
            if i - idx < res {
                res = i - idx
            }
        }
        queue = append(queue, i)
    }
    if res == int(1e9) {
        return -1
    }
    return res
}
