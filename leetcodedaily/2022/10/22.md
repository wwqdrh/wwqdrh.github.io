func jobScheduling(startTime []int, endTime []int, profit []int) int {
    idx := make([]int, len(startTime))
    for i := range idx {
        idx[i] = i
    }
    sort.Slice(idx, func(i, j int) bool { 
        l, r := idx[i], idx[j]
        return endTime[l] < endTime[r]
    })

    n := len(startTime)
    // dp[i]表示前i个！！注意不包含第i的最大报酬
    dp := make([]int, n+1)
    for i := 1; i <= n; i++ {
        k := sort.Search(i, func(j int) bool {
            return endTime[idx[j]] > startTime[idx[i-1]]
        })
        dp[i] = max(dp[i-1], dp[k]+profit[idx[i-1]]) // 不选这一个和要选这一个中选择
    }
    return dp[n]
}

func max(a, b int) int {
    if a > b {
        return a
    }
    return b
}
