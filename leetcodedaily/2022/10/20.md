func kthGrammar(n int, k int) int {
    // 树形dp 从底向上空间不够
    // 从底向下反证法求解

    guest := 1

    curguest := guest
    curk := k - 1 // 如果是奇数到了上一层需要翻转
    for ;n > 1; n-- {
        if curk & 1 == 1 {
            curguest = curguest ^ 1
        }
        curk = curk >> 1
    }
    if curguest == 0 {
        return guest
    } else {
        return 0
    }
}
