func partitionDisjoint(nums []int) int {
    leftMax, leftCur, curMax := nums[0], 0, nums[0]
    for i := 1; i < len(nums); i++ {
        if nums[i] > curMax {
            curMax = nums[i]
        }

        if leftMax > nums[i] {
            leftCur = i
            leftMax = curMax
        }
    }
    return leftCur + 1
}
