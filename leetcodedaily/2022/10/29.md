func countMatches(items [][]string, ruleKey string, ruleValue string) int {
    idx := -1
    switch ruleKey {
    case "type":
        idx = 0
    case "color":
        idx = 1
    case "name":
        idx = 2
    }
    if idx == -1 {
        return 0
    }

    res := 0
    for _, item := range items {
        if item[idx] == ruleValue {
            res++
        }
    }
    return res
}
