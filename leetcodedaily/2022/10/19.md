func countStudents(students, sandwiches []int) int {
    s1 := 0
    for _, v := range students {
        s1 += v
    }
    s0 := len(students) - s1
    for _, x := range sandwiches {
        if x == 0 && s0 > 0 {
            s0--
        } else if x == 1 && s1 > 0 {
            s1--
        } else {
            break
        }
    }
    return s0 + s1
}
