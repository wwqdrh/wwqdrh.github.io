func reformatNumber(number string) string {
    nums := []rune{}
    for _, ch := range number {
        if ch >= '0' && ch <= '9' {
            nums = append(nums, ch)
        }
    }

    res := []string{}
    i := 0
    n := len(nums)
    for i+4 < n {
        res = append(res, string(nums[i:i+3]))
        i += 3
    }

    switch n - i {
    case 1,2,3:
        res = append(res, string(nums[i:n]))
    case 4:
        res = append(res, string(nums[i:i+2]), string(nums[i+2:n]))
    }
    return strings.Join(res, "-")
}
