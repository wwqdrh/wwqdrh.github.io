func letterCasePermutation(s string) []string {
    res := []string{}
    
    sbytes := []rune(s)
    n := len(sbytes)
    var dfs func(int)
    dfs = func(idx int) {
        for idx < n && unicode.IsDigit(sbytes[idx]) {
            idx++
        }
        if idx == n {
            res = append(res, string(sbytes))
            return
        }

        dfs(idx+1)
        sbytes[idx] ^= 32
        dfs(idx+1)
        sbytes[idx] ^= 32
    }
    dfs(0)
    return res
}
