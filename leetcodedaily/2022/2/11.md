排序后 判断第i个和第i+k-1个

```go
func minimumDifference(nums []int, k int) int {
    sort.Slice(nums, func(i, j int) bool { return nums[i] < nums[j] })
	numsLen := len(nums)
	res := 1<<63 - 1
	for i := 0; i < numsLen; i++ {
		j := i + k - 1
		if j >= numsLen {
			break
		}
		if nums[j]-nums[i] < res {
			res = nums[j] - nums[i]
		}
	}
	return res
}
```
