func maxNumberOfBalloons(text string) int {
    cnt := [5]int{} // balon
    for _, ch := range text {
        if ch == 'b' {
            cnt[0]++
        } else if ch == 'a' {
            cnt[1]++
        } else if ch == 'l' {
            cnt[2]++
        } else if ch == 'o' {
            cnt[3]++
        } else if ch == 'n' {
            cnt[4]++
        }
    }
    cnt[2] /= 2
    cnt[3] /= 2
    ans := cnt[0]
    for _, item := range cnt {
        if item < ans {
            ans = item
        }
    }
    return ans
}
