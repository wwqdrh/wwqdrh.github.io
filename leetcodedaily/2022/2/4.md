简单遍历

```go
func countGoodRectangles(rectangles [][]int) int {
    maxItem := 0
    mapping := map[int]int{}
    for _, item := range rectangles {
        var c int
        if item[0] < item[1] {
            c = item[0]
        } else {
            c = item[1]
        }
        mapping[c]++
        if c > maxItem {
            maxItem = c
        }
    }
    return mapping[maxItem]
}
```
