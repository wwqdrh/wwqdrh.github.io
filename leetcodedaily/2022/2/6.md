计数 求和

```go
func sumOfUnique(nums []int) int {
    // 使用0 1 2 表示元素的访问状态 这样能够只需要一次遍历
    state := map[int]int{}
    res := 0
    for _, num := range nums {
        if state[num] == 0 {
            res += num
            state[num] = 1
        } else if state[num] == 1 {
            res -= num
            state[num] = 2
        }
    }
    return res
}
```
