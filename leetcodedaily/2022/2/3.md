如果最后一个小于目标值一定可以选

```go
func findMinFibonacciNumbers(k int) int {
    ans := 0
    f := []int{1, 1}
    for f[len(f)-1] < k {
        f = append(f, f[len(f)-1]+f[len(f)-2])
    }
    for i := len(f) - 1; k > 0; i-- {
        if k >= f[i] {
            k -= f[i]
            ans++
        }
    }
    return ans
}
```
