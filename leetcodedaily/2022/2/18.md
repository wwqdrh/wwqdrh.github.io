中间节点

```go
func findCenter(edges [][]int) int {
    // 计数 判断谁的个数是n-1
    cnt := map[int]int{}
    for _, item := range edges {
        cnt[item[0]]++
        cnt[item[1]]++
    }
    for key, val := range cnt {
        if val == len(edges) {
            return key
        }
    }
    return -1
}
```
