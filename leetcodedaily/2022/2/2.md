func reversePrefix(word string, ch byte) string {
    idx := strings.IndexByte(word, ch)
    if idx == -1 {
        return word
    }

    res := []byte{}
    for i := idx; i >= 0; i-- {
        res = append(res, word[i])
    }
    for i := idx+1; i < len(word); i++ {
        res = append(res, word[i])
    }
    return string(res)
}
