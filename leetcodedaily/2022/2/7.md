func longestDiverseString(a int, b int, c int) string {
    ans := []byte{}
	cnt := []struct {
		c  int
		ch byte
	}{
        {a, 'a'}, {b, 'b'}, {c, 'c'},
    }
	for {
		sort.Slice(cnt, func(i, j int) bool { return cnt[i].c > cnt[j].c })
		hasNext := false
		for i, item := range cnt {
			if item.c <= 0 {
				break
			}
			m := len(ans)
			if m >= 2 && ans[m-2] == item.ch && ans[m-1] == item.ch {
				continue
			}
			hasNext = true
			ans = append(ans, item.ch)
			cnt[i].c--
			break
		}
		if !hasNext {
			return string(ans)
		}
	}
}
