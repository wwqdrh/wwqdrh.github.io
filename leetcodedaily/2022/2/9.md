简单遍历

```go
func countKDifference(nums []int, k int) int {
    length := len(nums)
    res := 0
    for i := 0; i < length; i++ {
        for j := i + 1; j < length; j++ {
            if abs(nums[i], nums[j]) == k {
                res++
            }
        }
    }
    return res
}

func abs(a, b int) int {
    res := a - b
    if res < 0 {
        res = -res
    }
    return res
}
```
