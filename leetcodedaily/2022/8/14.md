func maxScore(s string) int {
    score := int('1' - s[0]) + strings.Count(s[1:], "1")
    ans := score
    for _, ch := range s[1:len(s)-1] {
        if ch == '1' {
            score -= 1
        } else {
            score += 1
        }
        if score > ans {
            ans = score
        }
    }
    return ans
}
