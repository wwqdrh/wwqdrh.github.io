找规律

如果k大于1，那么相当于冒泡，一定能够使其严格有序

```go
func orderlyQueue(s string, k int) string {
    if k == 1 {
        res := s
        for i, n := 0, len(s); i < n; i++ {
            cur := s[i:] + s[:i]
            if cur < res {
                res = cur
            }
        }
        return res
    } else {
        parts := []byte(s)
        sort.Slice(parts, func(i, j int) bool {
            return parts[i] < parts[j]
        })
        return string(parts)
    }
}
```
