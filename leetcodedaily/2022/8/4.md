要求取出的元素数量尽可能的少的前提下，要大于剩下的元素，取出的元素之和尽可能的大

贪心

降序排序

```go
func minSubsequence(nums []int) []int {
    sort.Sort(sort.Reverse(sort.IntSlice(nums)))
    tot := 0
    for _, num := range nums {
        tot += num
    }
    for i, sum := 0, 0; ; i++ {
        sum += nums[i]
        if sum > tot-sum {
            return nums[:i+1]
        }
    }
}
```
