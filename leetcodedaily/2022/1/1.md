下标的转换

```go
func construct2DArray(original []int, m int, n int) [][]int {
    if m * n != len(original) {
        return nil
    }

    res := make([][]int, m)
    for i := range res {
        res[i] = make([]int, n)
    }

    for i, item := range original {
        x, y := i / n, i % n
        res[x][y] = item
    }
    return res
}
```
