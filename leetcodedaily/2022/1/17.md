动态规划

第i个位置以a、e、i、o、u结尾的情况由前面的决定

由于只跟前面一个有关，所以可以存储空间进行优化

```go
func countVowelPermutation(n int) int {
	const mod int = 1e9+7
	dp := [5]int{1,1,1,1,1} // a e i o u
	for i := 1; i < n; i++ {
		dp = [5]int {
			(dp[1] + dp[2] + dp[4]) % mod,
			(dp[0] + dp[2]) % mod,
			(dp[1] + dp[3]) % mod,
			dp[2],
			(dp[2] + dp[3]) % mod,
		}
	}

	ans := 0
	for _, v := range dp {
		ans = (ans + v) % mod
	}
	return ans
}
```
