将`?`替换为其他字符

由于左右只有两种情况，所以这个`?`只需要替换为`(a, b, c)`中的一种即可


```go
func modifyString(s string) string {
    res := []byte(s)
    n := len(res)
    for i, ch := range res {
        if ch == '?' {
            for b := byte('a'); b <= 'c'; b++ {
                if !(i > 0 && res[i-1] == b || i < n-1 && res[i+1] == b) {
                    res[i] = b
                    break
                }
            }
        }
    }
    return string(res)
}
```
