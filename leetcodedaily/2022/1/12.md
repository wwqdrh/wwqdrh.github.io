简单遍历，记录可能的最小和第二小

```go
func increasingTriplet(nums []int) bool {
	if len(nums) < 3 {
		return false
	}
	first, second := nums[0], math.MaxInt64
	for i := 1; i < len(nums); i++ {
		cur := nums[i]
		if cur > second {
			return true
		} else if cur > first {
			second = cur
		} else {
			first = cur
		}
	}
	return false
}
```
