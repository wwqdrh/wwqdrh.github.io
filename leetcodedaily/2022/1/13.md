只需要第一大和第二大满足就行

```go
func dominantIndex(nums []int) int {
    if len(nums) == 0 {
        return -1
    }

    firstNum, secondNum := 0, 0
    id := 0
    for i, item := range nums {
        if item > firstNum {
            secondNum = firstNum
            firstNum = item
            id = i
        } else if item > secondNum {
            secondNum = item
        }
    }

    if secondNum * 2 > firstNum {
        return -1
    } else {
        return id
    }
}
```
