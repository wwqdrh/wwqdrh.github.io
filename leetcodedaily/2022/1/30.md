只在其中一个地方出现过，那其实就是计数为1的

```go
func uncommonFromSentences(s1 string, s2 string) []string {
    cnt := map[string]int{}
    for _, item := range strings.Fields(s1) {
        cnt[item]++
    }
    for _, item := range strings.Fields(s2) {
        cnt[item]++
    }
    
    res := []string{}
    for key, val := range cnt {
        if val == 1 {
            res = append(res, key)
        }
    }
    return res
}
```
