```go
func slowestKey(releaseTimes []int, keysPressed string) byte {
    maxTimes := releaseTimes[0]
    res := keysPressed[0]
    for i := 1; i < len(releaseTimes); i++ {
        curTime := releaseTimes[i] - releaseTimes[i-1]
        if curTime > maxTimes || (curTime == maxTimes && keysPressed[i] > res) {
            maxTimes = curTime
            res = keysPressed[i]
        }
    }
    return res
}
```
