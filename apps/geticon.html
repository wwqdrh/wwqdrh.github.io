parser = argparse.ArgumentParser(description="获取图标")
parser.add_argument("-a", dest="address", nargs="+", required=True)  # 地址
parser.add_argument("-s", dest="savepath", type=str, default=os.getcwd())
parser.add_argument("-n", dest="name", nargs="+", default="favcion")


def retry(num: int):
    def decroter(fn: Callable[..., Coroutine]):
        # @functools.partial(fn)
        async def _f(*args, **kwargs):
            curNum = num
            while curNum > 0:
                try:
                    res = await fn(*args, **kwargs)
                except Exception:
                    curNum -= 1
                else:
                    return res

        return _f

    return decroter


@retry(5)
async def get_content(url: str) -> httpx.Response:
    async with httpx.AsyncClient() as client:
        resp = await client.get(url, follow_redirects=True, timeout=10)
        if resp.status_code != 200:
            raise Exception("not 200")
        return resp


async def main(cmds: list[str] = None) -> list[str]:
    args = parser.parse_args(cmds)
    address, names = args.address, args.name
    n: int = len(address)  # type: ignore

    # 创建目录
    savepath = Path(args.savepath)
    if savepath.is_file():
        print("savepath是文件")
        exit(1)
    # args.address可能有多个
    if len(address) != len(names):
        print("参数个数不对")
        exit(1)

    if not savepath.exists():
        os.makedirs(savepath)

    try:
        return await asyncio.gather(
            *list(
                download(address[i], args.savepath, names[i])
                for i in range(len(address))
            ),
            return_exceptions=False,
        )
    except Exception as e:
        print(f"ERROR: {e}")
        return ["unknow.ico" for i in range(n)]


async def download(address: str, savepath: str, name: str) -> str:
    resp: httpx.Response | None = await get_content(address)
    if resp is None or resp.status_code != 200:
        return "unknown.ico"

    try:
        iconurl = resp.request.url.join(parse_iconurl(resp.text))
        return await download_favicon(str(iconurl), Path(savepath), name)
    except Exception as e:
        return "unknown.ico"


async def download_favicon(url: str, path: Path, name: str) -> str:
    ext = os.path.splitext(url)[1]
    resp: httpx.Response = await get_content(url)
    if resp is None or resp.status_code != 200:
        raise Exception("下载图标失败")

    iconpath = path / f"{name}{ext}"
    with open(iconpath, mode="wb") as f:
        f.write(resp.content)

    return os.path.basename(iconpath)


def parse_iconurl(content: str) -> str:
    return command1(content) or command2(content) or "favicon.ico"


def command1(data: str) -> str:
    items = re.findall(r'<link.*rel="icon".*>', data)
    urls = []
    for i in items:
        if (p := re.search(r'href="(.*?)"', i)) is None:
            continue
        url = p.group(1)
        urls.append(url)
    if urls:
        return urls[0]
    else:
        return ""


def command2(data: str) -> str:
    """
    <link rel="shortcut icon" href="/static/shared/icon/favicon.ico">
    """
    items = re.findall(r'<link.*rel="shortcut icon".*>', data)
    urls = []
    for i in items:
        if (p := re.search(r'href="(.*?)"', i)) is None:
            continue
        url = p.group(1)
        urls.append(url)
    if urls:
        return urls[0]
    else:
        return ""


if __name__ == "__main__":
    print(" ".join(asyncio.run(main())))
