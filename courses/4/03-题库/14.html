羊狼农夫过河

bfs，将各种情况抽象为状态，在运动物过河的时候考虑所有满足情况的条件，找到最小次数

羊、狼、农夫都在岸边，当羊的数量小于狼的数量时，狼会攻击羊，农夫则会损失羊。农夫有一艘容量固定的船，能够承载固定数量的动物。

要求求出不损失羊情况下将全部羊和狼运到对岸需要的最小次数。

只计算农夫去对岸的次数，回程时农夫不会运送羊和狼。

备注：农夫在或农夫离开后羊的数量大于狼的数量时狼不会攻击羊。

农夫自身不占用船的容量。

输入描述

第一行输入为M，N，X， 分别代表羊的数量，狼的数量，小船的容量。

输出描述

输出不损失羊情况下将全部羊和狼运到对岸需要的最小次数（若无法满足条件则输出0）。

package main

import "fmt"

type state struct {
	leftSheep  int
	leftWolf   int
	rightSheep int
	rightWolf  int
}

func (s state) IsValid() bool {
	return s.leftSheep > s.leftWolf &&
		s.leftSheep >= 0 && s.rightSheep >= 0 &&
		s.leftWolf >= 0 && s.rightWolf >= 0 &&
		s.rightSheep > s.rightWolf
}

func main() {
	var m, n, x int
	fmt.Scan(&m, &n, &x)

	initstate := state{m, n, 0, 0}
	goalstate := state{0, 0, m, n}
	fmt.Println(getStep(initstate, goalstate, x-1))
}

func getStep(ori state, target state, capacity int) int {
	queue := []state{ori}
	visit := map[state]bool{ori: true}

	steps := 0
	for len(queue) > 0 {
		newqueue := []state{}
		for _, curstate := range queue {
			// 农夫可以带capacity只动物，可以自由分配成i只羊或者j只狼，只要能够满足state合法
			for i := 0; i <= capacity; i++ {
				newstate := state{
					leftSheep:  curstate.leftSheep - i,
					leftWolf:   curstate.leftWolf - (capacity - i),
					rightSheep: curstate.rightSheep + i,
					rightWolf:  curstate.rightWolf + (capacity - i),
				}
				if _, ok := visit[newstate]; ok {
					continue
				}

				if newstate == target {
					return steps + 1
				}
				if newstate.IsValid() {
					newqueue = append(newqueue, newstate)
					visit[newstate] = true
				}
			}
		}
		queue = newqueue
		steps++
	}

	return -1
}

特异性双端队列

给定一个队列，但是这个队列比较特殊，可以从头部添加数据，也可以从尾部添加数据，但是只能从头部删除数据。 输入一个数字n，会依次添加数字1n（也就是添加n次）。 但是在添加数据的过程中，也会删除数据，要求删除必须按照1n按照顺序进行删除，所以在删除时，可以根据需要调整队列中数字的顺序以满足删除条件。

输入描述

第一行一个数据N，表示数据的范围。

接下来的2N行是添加和删除语句。 其中：

head add x 表示从头部添加元素 x，

tail add 表示从尾部添加元素，

remove表示删除元素。

输出描述

输出一个数字，表示最小的调整顺序次数。

package main

import "fmt"

func main() {
	var n int
	fmt.Scan(&n)

	queue := []int{}
	removeCount := 0
	adjustCount := 0

	for i := 1; i <= n; i++ {
		var op string
		var x int
		fmt.Scan(&op)

		if op == "head" {
			fmt.Scan(&op)
			fmt.Scan(&x)
			queue = append([]int{x}, queue...)
		} else if op == "tail" {
			fmt.Scan(&op)
			fmt.Scan(&x)
			queue = append(queue, x)
		} else {
			removeCount++
			if len(queue) > 0 && queue[0] == removeCount {
				queue = queue[1:]
			} else {
				for j := 1; j < len(queue); {
					// 如果队列中没有要删除的数字，需要调整队列顺序
					for k := 0; k < len(queue); k++ {
						if queue[k] == removeCount {
							queue = append(queue[:k], queue[k+1:]...)
							adjustCount++
							break
						}
					}
				}
			}
		}
	}
	fmt.Println(adjustCount)
}

package main

import (
	"bufio"
	"fmt"
	"os"
	"strings"
)

type trie struct {
	children [26]*trie
	isEnd    bool
}

func newTrie(words []string) *trie {
	root := &trie{}
	for _, word := range words {
		cur := root
		for _, ch := range word {
			x := ch - 'a'
			if cur.children[x] == nil {
				cur.children[x] = &trie{}
			}
			cur = cur.children[x]
		}
		cur.isEnd = true
	}
	return root
}

// 判断全路径下是否都是isend
func (t *trie) isAbsPath(p string) bool {
	cur := t
	for _, ch := range p {
		x := ch - 'a'
		if cur.children[x] == nil || !cur.children[x].isEnd {
			return false
		}
		cur = cur.children[x]
	}
	return true
}

func main() {
	input := bufio.NewScanner(os.Stdin)
	input.Scan()
	words := strings.Fields(input.Text())

	t := newTrie(words)

	maxLength := 0
	passpool := []string{}

	for _, word := range words {
		if t.isAbsPath(word) {
			if len(word) > maxLength {
				maxLength = len(word)
				passpool = []string{word}
			} else if len(word) == maxLength {
				passpool = append(passpool, word)
			}
		}
	}
	ans := ""
	if len(passpool) > 0 {
		ans = passpool[0]
		for _, item := range passpool[1:] {
			if item > ans {
				ans = item
			}
		}
	}
	fmt.Println(ans)
}

def max_friendly_score(F):
    # 找到所有空位的位置
    empty_positions = [i for i in range(len(F)) if F[i] == 0]

    # 如果没有空位，返回0
    if not empty_positions:
        return 0

    max_score = 0
    for pos in empty_positions:
        # 向左扩展
        left_score = 0
        i = pos - 1
        while i >= 0 and F[i] == 1:
            left_score += 1
            i -= 1

        # 向右扩展
        right_score = 0
        i = pos + 1
        while i < len(F) and F[i] == 1:
            right_score += 1
            i += 1

        # 计算友好度
        score = left_score + right_score
        if score > max_score:
            max_score = score

    return max_score

等和子数组最小和

给定一个数组nums，将元素分为若干个组，使得每组和相等，求出满足条件的所有分组中，组内元素和的最小值。

输入描述

第一行输入 m 接着输入m个数，表示此数组 数据范围：1<=M<=50, 1<=nums[i]<=50

输出描述

最小拆分数组和

示例1 输入输出示例仅供调试，后台判断数据一般不包含示例

输入

7 4 3 2 3 5 2 1

输出

5

说明：

可以等分的情况有：

4 个子集（5），（1,4），（2,3），（2,3）

2 个子集（5, 1, 4），（2,3, 2,3）

但最小的为5。


import (
	"fmt"
	"math"
	"sort"
)

func dfs(nums []int, target int, start int, used []bool, curSum int, curCnt int) bool {
	if curCnt == 0 {
		return true
	}
	if curSum == target {
		return dfs(nums, target, 0, used, 0, curCnt-1)
	}
	for i := start; i < len(nums); i++ {
		if !used[i] && curSum+nums[i] <= target {
			used[i] = true
			if dfs(nums, target, i+1, used, curSum+nums[i], curCnt) {
				return true
			}
			used[i] = false
		}
	}
	return false
}

func main() {
	var m int
	fmt.Scan(&m)
	nums := make([]int, m)
	for i := 0; i < m; i++ {
		fmt.Scan(&nums[i])
	}

	sort.Slice(nums, func(i, j int) bool {
		return nums[i] < nums[j]
	})
	totalSum := 0
	for _, num := range nums {
		totalSum += num
	}
	res := math.MaxInt32
	for i := 1; i <= m; i++ {
		if totalSum%i != 0 {
			continue
		}

		used := make([]bool, len(nums))
		if dfs(nums, totalSum/i, 0, used, 0, i) {
			if totalSum/i < res {
				res = totalSum / i
			}
		}
	}
	fmt.Println(res)
}

部门组织绿岛骑行团建活动。租用公共双人自行车，每辆自行车最多坐两人，最大载重M。 给出部门每个人的体重，请问最多需要租用多少双人自行车。

输入描述

第一行两个数字m、n，分别代表自行车限重，部门总人数。

第二行，n个数字，代表每个人的体重，体重都小于等于自行车限重m。

0<m<=200 0<n<=1000000

输出描述

最小需要的双人自行车数量。

示例1 输入输出示例仅供调试，后台判题数据一般不包含示例

输入

3 4 3 2 2 1

输出

3

def max_bikes(m, n, weights):
    # 将所有人按照体重从小到大排序
    weights.sort()

    # 使用双指针法寻找能够一起骑行的人
    left, right = 0, n - 1
    count = 0
    while left <= right:
        if weights[left] + weights[right] <= m:
            # 如果两个人的体重之和不超过自行车最大载重，就一起骑行
            left += 1
            right -= 1
        else:
            # 否则，只有重的人单独骑行
            right -= 1
        count += 1

    return count

小明在学习二进制时，发现了一类不含 101的数，也就是：

将数字用二进制表示，不能出现 101 。 现在给定一个整数区间 [l,r] ，请问这个区间包含了多少个不含 101 的数？

输入描述

输入的唯一一行包含两个正整数 l， r（ 1 ≤ l ≤ r ≤ 10^9）。

输出描述

输出的唯一一行包含一个整数，表示在 [l,r] 区间内一共有几个不含 101 的数。
