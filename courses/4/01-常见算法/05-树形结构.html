完全二叉树的节点个数

https://leetcode.cn/problems/count-complete-tree-nodes/

给定一个完全二叉树的根节点，求这个树的节点个数

func countNodes(root *TreeNode) int {
    if root == nil {
        return 0
    }
    level := 0
    for node := root; node.Left != nil; node = node.Left {
        level++
    }
    return sort.Search(1<<(level+1), func(k int) bool {
        if k <= 1<<level {
            return false
        }
        bits := 1 << (level - 1)
        node := root
        for node != nil && bits > 0 {
            if bits&k == 0 {
                node = node.Left
            } else {
                node = node.Right
            }
            bits >>= 1
        }
        return node == nil
    }) - 1
}

不同的二叉搜索树

https://leetcode.cn/problems/unique-binary-search-trees/

与不同的二叉搜索树2类似，这里只需要求有多少种类

func numTrees(n int) int {
    G := make([]int, n + 1)
    G[0], G[1] = 1, 1
    for i := 2; i <= n; i++ {
        for j := 1; j <= i; j++ {
            G[i] += G[j-1] * G[i-j]
        }
    }
    return G[n]
}

验证二叉搜索树

https://leetcode.cn/problems/validate-binary-search-tree/

给一个二叉树的根节点，判断这个二叉树是否是合法的

func isValidBST(root *TreeNode) bool {
    return helper(root, math.MinInt64, math.MaxInt64)
}

func helper(root *TreeNode, lower, upper int) bool {
    if root == nil {
        return true
    }
    if root.Val <= lower || root.Val >= upper {
        return false
    }
    return helper(root.Left, lower, root.Val) && helper(root.Right, root.Val, upper)
}

恢复二叉搜索树

https://leetcode.cn/problems/recover-binary-search-tree/

给定一个二叉树的根节点，这个二叉树中恰好有两个节点的值被错误的交换，在不改变其结构的情况下，恢复这颗树

func recoverTree(root *TreeNode)  {
    stack := []*TreeNode{}
    var x, y, pred *TreeNode
    for len(stack) > 0 || root != nil {
        for root != nil {
            stack = append(stack, root)
            root = root.Left
        }
        root = stack[len(stack)-1]
        stack = stack[:len(stack)-1]
        if pred != nil && root.Val < pred.Val {
            y = root
            if x == nil {
                x = pred
            } else {
                break
            }
        }
        pred = root
        root = root.Right
    }
    x.Val, y.Val = y.Val, x.Val
}

路径总和2

https://leetcode.cn/problems/path-sum-ii/

给定一个二叉树，找到路径和为目标值的这条路径

func pathSum(root *TreeNode, targetSum int) (ans [][]int) {
    path := []int{}
    var dfs func(*TreeNode, int)
    dfs = func(node *TreeNode, left int) {
        if node == nil {
            return
        }
        left -= node.Val
        path = append(path, node.Val)
        defer func() { path = path[:len(path)-1] }()
        if node.Left == nil && node.Right == nil && left == 0 {
            ans = append(ans, append([]int(nil), path...))
            return
        }
        dfs(node.Left, left)
        dfs(node.Right, left)
    }
    dfs(root, targetSum)
    return
}

填充每个节点的下一个右侧节点指针

https://leetcode.cn/problems/populating-next-right-pointers-in-each-node/

给定一个完美二叉树，填充这个二叉树节点中的next指针，指向节点的右继节点

func connect(root *Node) *Node {
    if root == nil {
        return root
    }

    // 每次循环从该层的最左侧节点开始
    for leftmost := root; leftmost.Left != nil; leftmost = leftmost.Left {
        // 通过 Next 遍历这一层节点，为下一层的节点更新 Next 指针
        for node := leftmost; node != nil; node = node.Next {
            // 左节点指向右节点
            node.Left.Next = node.Right

            // 右节点指向下一个左节点
            if node.Next != nil {
                node.Right.Next = node.Next.Left
            }
        }
    }

    // 返回根节点
    return root
}

二叉搜索树中第k小的元素

https://leetcode.cn/problems/kth-smallest-element-in-a-bst/

给定一个二叉搜索树节点和整数k，找到第k个最小元素

func kthSmallest(root *TreeNode, k int) int {
    stack := []*TreeNode{}
    for {
        for root != nil {
            stack = append(stack, root)
            root = root.Left
        }
        stack, root = stack[:len(stack)-1], stack[len(stack)-1]
        k--
        if k == 0 {
            return root.Val
        }
        root = root.Right
    }
}

二叉搜索树的最近公共祖先

https://leetcode.cn/problems/lowest-common-ancestor-of-a-binary-search-tree/

给定一个二叉搜索树, 找到该树中两个指定节点的最近公共祖先。

func lowestCommonAncestor(root, p, q *TreeNode) (ancestor *TreeNode) {
    ancestor = root
    for {
        if p.Val < ancestor.Val && q.Val < ancestor.Val {
            ancestor = ancestor.Left
        } else if p.Val > ancestor.Val && q.Val > ancestor.Val {
            ancestor = ancestor.Right
        } else {
            return
        }
    }
}

不同的二叉搜索树2

https://leetcode.cn/problems/unique-binary-search-trees-ii/

返回由n个节点组成且节点值从1到n互不相同的不同二叉搜索树，可以按照任意顺序返回答案

func generateTrees(n int) []*TreeNode {
    if n == 0 {
        return nil
    }
    return helper(1, n)
}

func helper(start, end int) []*TreeNode {
    if start > end {
        return []*TreeNode{nil}
    }
    allTrees := []*TreeNode{}
    // 枚举可行根节点
    for i := start; i <= end; i++ {
        // 获得所有可行的左子树集合
        leftTrees := helper(start, i - 1)
        // 获得所有可行的右子树集合
        rightTrees := helper(i + 1, end)
        // 从左子树集合中选出一棵左子树，从右子树集合中选出一棵右子树，拼接到根节点上
        for _, left := range leftTrees {
            for _, right := range rightTrees {
                currTree := &TreeNode{i, nil, nil}
                currTree.Left = left
                currTree.Right = right
                allTrees = append(allTrees, currTree)
            }
        }
    }
    return allTrees
}

给定一个二叉树，填充这个二叉树节点中的next指针，指向节点的右继节点，如果不存在则将其设置为NULL

func connect(root *Node) *Node {
	if root == nil {return nil}
    q := []*Node{root}
    for len(q) > 0 {
        qlen := len(q)
        pre := &Node{}

        for i := 0; i < qlen; i++ {
            // 把q连接为list
            n := q[0]
            q = q[1:] // 此两行为q.pop()
            if pre != nil {pre.Next = n} // pre == nil 即为本层第一个节点
            pre = n

            if n.Left != nil {q = append(q, n.Left)}
            if n.Right != nil {q = append(q, n.Right)}
        }
    }
    return root
}

二叉树的层序遍历

https://leetcode.cn/problems/binary-tree-level-order-traversal/

给定一个二叉树，返回该二叉树层序遍历获得的值

func levelOrder(root *TreeNode) [][]int {
    ret := [][]int{}
    if root == nil {
        return ret
    }
    q := []*TreeNode{root}
    for i := 0; len(q) > 0; i++ {
        ret = append(ret, []int{})
        p := []*TreeNode{}
        for j := 0; j < len(q); j++ {
            node := q[j]
            ret[i] = append(ret[i], node.Val)
            if node.Left != nil {
                p = append(p, node.Left)
            }
            if node.Right != nil {
                p = append(p, node.Right)
            }
        }
        q = p
    }
    return ret
}

二叉树的锯齿形层序遍历

https://leetcode.cn/problems/binary-tree-zigzag-level-order-traversal/

给定一个二叉树，返回该二叉树层序遍历获得的值，每一层遍历方向是交叉进行的

func zigzagLevelOrder(root *TreeNode) (ans [][]int) {
    if root == nil {
        return
    }
    queue := []*TreeNode{root}
    for level := 0; len(queue) > 0; level++ {
        vals := []int{}
        q := queue
        queue = nil
        for _, node := range q {
            vals = append(vals, node.Val)
            if node.Left != nil {
                queue = append(queue, node.Left)
            }
            if node.Right != nil {
                queue = append(queue, node.Right)
            }
        }
        // 本质上和层序遍历一样，我们只需要把奇数层的元素翻转即可
        if level%2 == 1 {
            for i, n := 0, len(vals); i < n/2; i++ {
                vals[i], vals[n-1-i] = vals[n-1-i], vals[i]
            }
        }
        ans = append(ans, vals)
    }
    return
}

二叉树的层序遍历2

https://leetcode.cn/problems/binary-tree-level-order-traversal-ii/

给定一个二叉树，返回从底向上层序遍历的结果集

func levelOrderBottom(root *TreeNode) [][]int {
    levelOrder := [][]int{}
    if root == nil {
        return levelOrder
    }
    queue := []*TreeNode{}
    queue = append(queue, root)
    for len(queue) > 0 {
        level := []int{}
        size := len(queue)
        for i := 0; i < size; i++ {
            node := queue[0]
            queue = queue[1:]
            level = append(level, node.Val)
            if node.Left != nil {
                queue = append(queue, node.Left)
            }
            if node.Right != nil {
                queue = append(queue, node.Right)
            }
        }
        levelOrder = append(levelOrder, level)
    }
    for i := 0; i < len(levelOrder) / 2; i++ {
        levelOrder[i], levelOrder[len(levelOrder) - 1 - i] = levelOrder[len(levelOrder) - 1 - i], levelOrder[i]
    }
    return levelOrder
}

红黑树

红黑树是一种自平衡的二叉查找树，它具有以下特点：

  • 每个节点要么是红色，要么是黑色。
  • 根节点是黑色的。
  • 每个叶子节点（NIL节点）是黑色的。
  • 如果一个节点是红色的，则它的两个子节点都是黑色的。
  • 从任意节点到其每个叶子节点的所有路径都包含相同数目的黑色节点。
  • 新插入的节点默认为红色。

自平衡性：红黑树通过对节点的颜色和旋转操作来保持树的平衡，在最坏情况下，树的高度不会超过2log(N+1)，其中N是树中节点的数量。

查找、插入、删除操作的时间复杂度都是O(logN)，在实际应用中，红黑树的效率比一般的平衡二叉树要高。

在红黑树中，值最小的节点一定是在最左边的叶子节点上。因此，寻找值最小的节点可以通过遍历红黑树的左子树来实现。

package main

import (
    "fmt"
)

//定义红黑树节点
type node struct {
    key int
    value string
    color bool // true: red, false: black
    left *node
    right *node
}

//定义红黑树
type RedBlackTree struct {
    root *node
}

//左旋函数
func (t *RedBlackTree) leftRotate(x *node) {
    y := x.right
    x.right, y.left = y.left, x
    if x == t.root {
        t.root = y
    }
}

//右旋函数
func (t *RedBlackTree) rightRotate(x *node) {
    y := x.left
    x.left, y.right = y.right, x
    if x == t.root {
        t.root = y
    }
}

//颜色翻转函数
func (t *RedBlackTree) flipColors(x *node) {
    x.color = !x.color
    x.left.color = !x.left.color
    x.right.color = !x.right.color
}

//插入节点函数
func (t *RedBlackTree) insert(key int, value string) {
    t.root = t.insertNode(t.root, key, value)
    t.root.color = false
}

func (t *RedBlackTree) insertNode(root *node, key int, value string) *node {
    if root == nil {
        return &node{key, value, true, nil, nil}
    }
    if key < root.key {
        root.left = t.insertNode(root.left, key, value)
    } else if key > root.key {
        root.right = t.insertNode(root.right, key, value)
    } else {
        root.value = value
    }

    //旋转和颜色调整
    if root.right != nil && root.right.color && root.left == nil {
        t.leftRotate(root)
        t.flipColors(root)
    }
    if root.left != nil && root.left.color && root.left.left != nil && root.right != nil && !root.right.color {
        t.rightRotate(root)
        t.flipColors(root)
    }
    if root.left != nil && root.left.color && root.left.right != nil && root.right != nil && root.right.color {
        t.leftRotate(root.left)
        t.rightRotate(root)
        t.flipColors(root)
    }

    return root
}

//查找节点函数
func (t *RedBlackTree) search(key int) string {
    current := t.root
    for current != nil {
        if current.key == key {
            return current.value
        } else if current.key > key {
            current = current.left
        } else {
            current = current.right
        }
    }
    return ""
}

//打印红黑树函数
func (t *RedBlackTree) printTree() {
    if t.root == nil {
        fmt.Println("Tree is empty")
    } else {
        t.printNode(t.root, "", true)
    }
}

func (t *RedBlackTree) printNode(node *node, prefix string, isLeft bool) {
    if node == nil {
        return
    }

    if isLeft {
        fmt.Printf("%v├──", prefix)
    } else {
        fmt.Printf("%v└──", prefix)
    }

    color := "B"
    if node.color {
        color = "R"
    }
    fmt.Printf("[%v %v]\n", color, node.value)

    t.printNode(node.left, prefix + "│   ", true)
    t.printNode(node.right, prefix + "    ", false)
}

func main() {
    t := &RedBlackTree{}
    t.insert(5, "value1")
    t.insert(4, "value2")
    t.insert(6, "value3")
    t.insert(3, "value4")
    t.insert(8, "value5")
    t.insert(7, "value6")
}

最小高度树

https://leetcode.cn/problems/minimum-height-trees/

给定一个包含n个节点的树(没有简单环路的连通图都是一棵树)。选择其中的一个点作为根节点，找到最小高度的树，返回树的根节点

func findMinHeightTrees(n int, edges [][]int) []int {
    if n == 1 {
        return []int{0}
    }

    g := make([][]int, n)
    for _, e := range edges {
        x, y := e[0], e[1]
        g[x] = append(g[x], y)
        g[y] = append(g[y], x)
    }

    parents := make([]int, n)
    bfs := func(start int) (x int) {
        vis := make([]bool, n)
        vis[start] = true
        q := []int{start}
        for len(q) > 0 {
            x, q = q[0], q[1:]
            for _, y := range g[x] {
                if !vis[y] {
                    vis[y] = true
                    parents[y] = x
                    q = append(q, y)
                }
            }
        }
        return
    }
    x := bfs(0) // 找到与节点 0 最远的节点 x
    y := bfs(x) // 找到与节点 x 最远的节点 y

    path := []int{}
    parents[x] = -1
    for y != -1 {
        path = append(path, y)
        y = parents[y]
    }
    m := len(path)
    if m%2 == 0 {
        return []int{path[m/2-1], path[m/2]}
    }
    return []int{path[m/2]}
}
