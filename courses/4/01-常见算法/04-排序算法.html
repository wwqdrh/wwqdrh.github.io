选择排序

每次选择所有待排序数据中最大的，然后再选择剩下的

时间复杂度稳定，运算次数为n-1 + n-2 + ... + 1 => (1/2) * n * (n-1), 最好和最坏的复杂度都是都是O(n^2)

func selectSort(array []int) []int  {
    n := len(array)
    for i:=0;i<n;i++ {
        index := i // 假设元素最小
        for j:=i+1;j<n-1;j++ {
            if (array[index] > array[j+1]) {
                index = j+1 // 记录元素最小的索引
            }
        }
        // 将最小的元素排序到左边
        array[i],array[index] = array[index],array[i]
        //i++,已经将最小的元素排序到最左边，然后在找第二大的元素
    }
    return array
}

#include <iostream>
using namespace std;

// 交换两个元素的值
void swap(int &a, int &b) {
    int temp = a;
    a = b;
    b = temp;
}

// 选择排序函数，参数是一个整数数组和它的长度
void selectSort(int array[], int n) {
    // 外层循环，从0到n-2，表示已排序部分的末尾
    for (int i = 0; i < n - 1; i++) {
        // 假设当前位置是最小元素的位置
        int index = i;
        // 内层循环，从i+1到n-1，表示未排序部分
        for (int j = i + 1; j < n; j++) {
            // 如果找到比当前最小元素更小的元素，更新最小元素的位置
            if (array[j] < array[index]) {
                index = j;
            }
        }
        // 如果最小元素不在当前位置，交换它们
        if (index != i) {
            swap(array[i], array[index]);
        }
    }
}

冒泡排序

从第一个元素开始不断发生比较并交换，在某一次冒泡过程中没有发生交换，说明整个数据已经有序了，那么冒泡就结束了

最好的情况是数据已经有序，只需要比较第一次，时间复杂度是O(n)；最坏的情况与选择排序一样；平均复杂度是趋于最坏情况

func bubbleSort(nums []int) []int {
    if len(nums) <= 1 {
        return nums
    }
    // 冒泡排序核心实现代码
    for i := 0; i < len(nums); i++ {
        flag := false
        for j := 0; j < len(nums) - i - 1; j++ {
            if nums[j] > nums[j+1] {
                nums[j], nums[j+1] = nums[j+1], nums[j]
                flag = true
            }
        }
        if !flag {
            break
        }
    }
    return nums
}

#include <iostream>
using namespace std;

void swap(int &a, int &b) {
    int temp = a;
    a = b;
    b = temp;
}

void bubbleSort(int array[], int n) {
    for (int i = 0; i < n - 1; i++) {
        bool swapped = false;
        for (int j = 0; j < n - i - 2; j++) {
            if (array[j] > array[j + 1]) {
                swap(array[j], array[j + 1]);
                swapped = true;
            }
        }
        if (!swapped) {
            break;
        }
    }
}

快速排序

每次选择一个作为基准(通常选择第一个)，然后将比基准小的放在前面，比基准大的放在后面。对于新分成的两个部分采用递归结构继续划分，直到数组长度为1不再划分

最好的情况，计算次数是n + 2*n/2 + 4*n/4 + ... + n*1, 这里总共能够分的层数是log2(n), 所以最好情况下时间复杂度是O(nlogn)

最坏的情况，每次选择所有的元素都在基准的后面(就是说这个数据本身已经是有序的了), 这时候计算次数是n-1 + n-2 + ... + 1, 时间复杂度是O(n^2)

不过与冒泡不同的是，快排的平均时间复杂度是趋于最好的情况的

func quickSort(nums []int, p int, r int) {
    // 递归终止条件
    if p >= r {
        return
    }
    // 获取分区位置
    q := partition(nums, p, r)
    // 递归分区（排序是在定位 pivot 的过程中实现的）
    quickSort(nums, p, q - 1)
    quickSort(nums, q + 1, r)
}
// 定位 pivot
func partition(nums []int, p int, r int) int {
    // 以当前数据序列最后一个元素作为初始 pivot
    pivot := nums[r]
    // 初始化 i、j 下标
    i := p
    // 后移 j 下标的遍历过程
    for j := p; j < r; j++ {
        // 将比 pivot 小的数丢到 [p...i-1] 中，剩下的 [i...j] 区间都是比 pivot 大的
        if nums[j] < pivot {
            // 互换 i、j 下标对应数据
            nums[i], nums[j] = nums[j], nums[i]
            // 将 i 下标后移一位
            i++
        }
    }
    // 最后将 pivot 与 i 下标对应数据值互换
    // 这样一来，pivot 就位于当前数据序列中间，i 也就是 pivot 值对应的下标
    nums[i], nums[r] = pivot, nums[i]
    // 返回 i 作为 pivot 分区位置
    return i
}

#include <iostream>
using namespace std;

// 定位 pivot 的函数，参数是一个整数数组，以及起始和结束的下标
int partition(int array[], int start, int end) {
    // 选择最后一个元素作为 pivot
    int pivot = array[end];
    // 初始化 i 和 j 的下标
    i = start;
    j = end - 1;
    // 循环遍历数组，直到 i 和 j 相遇
    while (i <= j) {
        // 如果左边的元素小于等于 pivot，就把 i 后移一位
        if (array[i] <= pivot) {
            i++;
        }
        // 如果右边的元素大于等于 pivot，就把 j 前移一位
        else if (array[j] >= pivot) {
            j--;
        }
        // 否则，就交换左右两边的元素
        else {
            swap(array[i], array[j]);
            i++;
            j--;
        }
    }
    // 最后，把 pivot 和 i 位置的元素交换
    swap(array[i], array[end]);
    // 返回 i 作为 pivot 的位置
    return i;
}

// 快速排序函数，参数是一个整数数组，以及起始和结束的下标
void quickSort(int array[], int start, int end) {
    // 如果起始下标大于等于结束下标，就返回
    if (start >= end) {
        return;
    }
    // 调用 partition 函数，获取 pivot 的位置
    int p = partition(array, start, end);
    // 对左边的部分递归地进行快速排序
    quickSort(array, start, p - 1);
    // 对右边的部分递归地进行快速排序
    quickSort(array, p + 1, end);
}

归并排序

直接分组而不是像选择排序那样根据大小关系进行分组，然后合并，这样计算次数稳定，时间复杂度稳定在O(nlog(n))

func mergeSort(nums []int) []int {
    if len(nums) <= 1 {
        return nums
    }
    // 获取分区位置
    p := len(nums) / 2
    // 通过递归分区
    left := mergeSort(nums[0:p])
    right := mergeSort(nums[p:])
    // 排序后合并
    return merge(left, right)
}
// 排序合并
func merge(left []int, right []int) []int {
    i, j := 0, 0
    m, n := len(left), len(right)
    // 用于存放结果集
    var result []int
    for i < m && i < n {
        if left[i] <= right[j] {
            result = append(result, left[i])
            i++
        } else {
            result = append(result, right[j])
            j++
        }
    }
    // 如果左侧区间还没有遍历完，将剩余数据放到结果集
    if i != m {
        result = append(result, left[i:]...)
    }
    // 如果右侧区间还没有遍历完，将剩余数据放到结果集
    if j != n {
        result = append(result, right[j:]...)
    }
    // 返回排序后的结果集
    return result
}

// 归并排序函数
void mergeSort(int array[], int left, int right) {
    if (left >= right) {
        return;
    }
    // 获取分区位置
    int mid = left + (right - left) / 2;
    // 通过递归分区
    mergeSort(array, left, mid);
    mergeSort(array, mid + 1, right);
    // 排序后合并
    merge(array, left, mid, right);
}

// 排序合并函数
void merge(int array[], int left, int mid, int right) {
    int i = left; // 左侧区间的指针
    int j = mid + 1; // 右侧区间的指针
    int k = 0; // 临时数组的指针
    int size = right - left + 1; // 临时数组的大小
    // 用于存放结果集的临时数组
    int* result = new int[size];
    while (i <= mid && j <= right) {
        if (array[i] <= array[j]) {
            result[k++] = array[i++];
        } else {
            result[k++] = array[j++];
        }
    }
    // 如果左侧区间还没有遍历完，将剩余数据放到结果集
    while (i <= mid) {
        result[k++] = array[i++];
    }
    // 如果右侧区间还没有遍历完，将剩余数据放到结果集
    while (j <= right) {
        result[k++] = array[j++];
    }
    // 将结果集复制回原数组
    for (int i = 0; i < size; i++) {
        array[left + i] = result[i];
    }
    // 释放临时数组的内存空间
    delete[] result;
}
