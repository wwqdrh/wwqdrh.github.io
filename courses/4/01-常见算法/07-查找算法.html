解决冲突的方法: 开放式寻址（线性探测、二次探测和双倍散列）和封闭式寻址（分离链接法）

在线性探测冲突解决方法中，每当发生冲突时，我们会一次扫描一个索引，以查找下一个空/被删除的插槽（当我们到达最后一个插槽时再重新环绕）。

二次探测。二次探测（Quadratic Probing）能够快速解决我们之前对线性探测的主要聚类问题，但是有可能存在空位但是这种方式计算不到，导致一直无法插入

双倍散列

h（v）//基地址 （h（v）+ 1 * h2（v））％M //第一个探测步骤，如果有碰撞 （h（v）+ 2 * h2（v））％M //第2次探测步骤，如果仍有冲突 （h（v）+ 3 * h2（v））％M //第三次探测步骤，如果仍有冲突 ... （h（v）+ k * h2（v））％M //第k个探测步骤等... 就是这样，探测器根据第二个散列函数h2（v）的值跳转，根据需要环绕散列表。

import "container/list"

type LRUCache struct {
    capacity int
    cache    map[int]*list.Element
    list     *list.List
}

type entry struct {
    key   int
    value int
}

func Constructor(capacity int) LRUCache {
    return LRUCache{
        capacity: capacity,
        cache:    make(map[int]*list.Element),
        list:     list.New(),
    }
}

func (this *LRUCache) Get(key int) int {
    if elem, ok := this.cache[key]; ok {
        this.list.MoveToFront(elem)
        return elem.Value.(*entry).value
    }
    return -1
}

func (this *LRUCache) Put(key int, value int) {
    if elem, ok := this.cache[key]; ok {
        elem.Value.(*entry).value = value
        this.list.MoveToFront(elem)
        return
    }
    if this.list.Len() == this.capacity {
        lastElem := this.list.Back()
        delete(this.cache, lastElem.Value.(*entry).key)
        this.list.Remove(lastElem)
    }
    newElem := this.list.PushFront(&entry{key: key, value: value})
    this.cache[key] = newElem
}

package lfu

import (
    "container/list"
)

type Cache struct {
    size     int
    capacity int
    items    map[interface{}]*list.Element
    freqs    map[int]*list.List
    minFreq  int
}

type CacheItem struct {
    key  interface{}
    val  interface{}
    freq int
}

func NewCache(capacity int) *Cache {
    return &Cache{
        size:     0,
        capacity: capacity,
        items:    make(map[interface{}]*list.Element),
        freqs:    make(map[int]*list.List),
        minFreq:  0,
    }
}

func (c *Cache) Get(key interface{}) interface{} {
    item, found := c.items[key]
    if !found {
        return nil
    }

    c.update(item.Value.(*CacheItem))

    return item.Value.(*CacheItem).val
}

func (c *Cache) Set(key interface{}, val interface{}) {
    if c.capacity == 0 {
        return
    }

    if item, found := c.items[key]; found {
        item.Value.(*CacheItem).val = val
        c.update(item.Value.(*CacheItem))
        return
    }

    if c.size == c.capacity {
        c.evict()
    }

    item := &CacheItem{
        key:  key,
        val:  val,
        freq: 1,
    }

    if list, found := c.freqs[1]; found {
        c.items[key] = list.PushFront(item)
    } else {
        list := list.New()
        c.freqs[1] = list
        c.items[key] = list.PushFront(item)
    }

    c.size++
    c.minFreq = 1
}

func (c *Cache) update(item *CacheItem) {
    freq := item.freq

    c.freqs[freq].Remove(c.items[item.key])

    if freq == c.minFreq && c.freqs[freq].Len() == 0 {
        c.minFreq++
    }

    item.freq++
    freq = item.freq

    if list, found := c.freqs[freq]; found {
        c.items[item.key] = list.PushFront(item)
    } else {
        list := list.New()
        c.freqs[freq] = list
        c.items[item.key] = list.PushFront(item)
    }
}

func (c *Cache) evict() {
    list := c.freqs[c.minFreq]

    item := list.Back().Value.(*CacheItem)

    list.Remove(c.items[item.key])

    delete(c.items, item.key)

    c.size--
}

func canJUmp(nums []int) bool {
	if len(nums)<=1{
		return true
	}
	dp:=make([]bool,len(nums))
	dp[0]=true
	for i:=1;i<len(nums);i++{
		for j:=i-1;j>=0;j--{
			if dp[j]&&nums[j]+j>=i{
				dp[i]=true
				break
			}
		}
	}
	return dp[len(nums)-1]
}

https://leetcode.cn/problems/unique-paths-ii/

给定一个m*n的二维矩阵，其中包含障碍物不能通过，问从左上到右下角有多少种不同的路径

func uniquePathsWithObstacles(obstacleGrid [][]int) int {
    n, m := len(obstacleGrid), len(obstacleGrid[0])
    f := make([]int, m)
    if obstacleGrid[0][0] == 0 {
        f[0] = 1
    }
    for i := 0; i < n; i++ {
        for j := 0; j < m; j++ {
            if obstacleGrid[i][j] == 1 {
                f[j] = 0
                continue
            }
            if j - 1 >= 0 && obstacleGrid[i][j-1] == 0 {
                f[j] += f[j-1]
            }
        }
    }
    return f[len(f)-1]
}
