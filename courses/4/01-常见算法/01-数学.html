最大数

https://leetcode.cn/problems/largest-number/

给定一组非负整数 nums，重新排列每个数的顺序（每个数不可拆分）使之组成一个最大的整数。

注意：输出结果可能非常大，所以你需要返回一个字符串而不是整数。

func largestNumber(nums []int) string {
    sort.Slice(nums, func(i, j int) bool {
        x, y := nums[i], nums[j]
        sx, sy := 10, 10
        for sx <= x {
            sx *= 10
        }
        for sy <= y {
            sy *= 10
        }
        return sy*x+y > sx*y+x
    })
    if nums[0] == 0 {
        return "0"
    }
    ans := []byte{}
    for _, x := range nums {
        ans = append(ans, strconv.Itoa(x)...)
    }
    return string(ans)
}

#include <algorithm>
#include <string>
#include <vector>

std::string largestNumber(std::vector<int> nums) {
    std::sort(nums.begin(), nums.end(), [](int x, int y) {
        int sx = 10, sy = 10;
        while (sx <= x) sx *= 10;
        while (sy <= y) sy *= 10;
        return sy*x + y > sx*y + x;
    });

    if (nums[0] == 0) return "0";
    std::string ans;
    for (int x : nums) {
        ans += std::to_string(x);
    }
    return ans;
}

整数替换

https://leetcode.cn/problems/integer-replacement/

给定一个正整数 n ，你可以做如下操作：

  • 如果 n 是偶数，则用 n / 2替换 n 。
  • 如果 n 是奇数，则可以用 n + 1或n - 1替换 n 。

返回 n 变为 1 所需的 最小替换次数 。

func integerReplacement(n int) (ans int) {
    for n != 1 {
        switch {
        case n%2 == 0:
            ans++
            n /= 2
        case n%4 == 1:
            ans += 2
            n /= 2
        case n == 3:
            ans += 2
            n = 1
        default:
            ans += 2
            n = n/2 + 1
        }
    }
    return
}

int integerReplacement(n int) {
    int cnt = 0;
    while (n > 1) {
        if (n % 2 == 0) {
            n /= 2;
        } else if (n == 3 || (n - 1) % 4 == 0) {
            n -= 1;
        } else {
            n += 1;
        }
        count++;
    }
    return count;
}

移掉k位数字

https://leetcode.cn/problems/remove-k-digits/

给你一个以字符串表示的非负整数 num 和一个整数 k ，移除这个数中的 k 位数字，使得剩下的数字最小。请你以字符串形式返回这个最小的数字。

func removeKdigits(num string, k int) string {
    stack := []byte{}
    for i := range num {
        digit := num[i]
        for k > 0 && len(stack) > 0 && digit < stack[len(stack)-1] {
            stack = stack[:len(stack)-1]
            k--
        }
        stack = append(stack, digit)
    }
    stack = stack[:len(stack)-k]
    ans := strings.TrimLeft(string(stack), "0")
    if ans == "" {
        ans = "0"
    }
    return ans
}

#include <string>
#include <stack>

std::string removeKdigits(std::string num, int k) {
    std::stack<char> stack;
    for (char digit : num) {
        while (k > 0 && !stack.empty() && digit < stack.top()) {
            stack.pop();
            k--;
        }
        stack.push(digit);
    }
    stack.resize(num.length() - k);

    std::string ans(stack.begin(), stack.end());
    ans.erase(0, ans.find_first_not_of('0'));

    if (ans.empty()) {
        return "0";
    }

    return ans;
}

水壶问题

https://leetcode.cn/problems/water-and-jug-problem/

给定两个不同容量的水壶，问是否能够通过这两个水壶准确得到target升水

class Solution:
    def canMeasureWater(self, x: int, y: int, z: int) -> bool:
        stack = [(0, 0)]
        self.seen = set()
        while stack:
            remain_x, remain_y = stack.pop()
            if remain_x == z or remain_y == z or remain_x + remain_y == z:
                return True
            if (remain_x, remain_y) in self.seen:
                continue
            self.seen.add((remain_x, remain_y))
            # 把 X 壶灌满。
            stack.append((x, remain_y))
            # 把 Y 壶灌满。
            stack.append((remain_x, y))
            # 把 X 壶倒空。
            stack.append((0, remain_y))
            # 把 Y 壶倒空。
            stack.append((remain_x, 0))
            # 把 X 壶的水灌进 Y 壶，直至灌满或倒空。
            stack.append((remain_x - min(remain_x, y - remain_y), remain_y + min(remain_x, y - remain_y)))
            # 把 Y 壶的水灌进 X 壶，直至灌满或倒空。
            stack.append((remain_x + min(remain_y, x - remain_x), remain_y - min(remain_y, x - remain_x)))
        return False

#include <algorithm>
#include <unordered_set>
#include <vector>

bool canMeasureWater(int x, int y, int z) {
    std::vector<std::pair<int, int>> stack;
    stack.push_back({0, 0});

    std::unordered_set<std::pair<int, int>> seen;

    while (!stack.empty()) {
        auto [remain_x, remain_y] = stack.back();
        stack.pop_back();

        if (remain_x == z || remain_y == z || remain_x + remain_y == z) {
            return true;
        }

        if (seen.count({remain_x, remain_y}) > 0) {
            continue;
        }

        seen.insert({remain_x, remain_y});
        stack.push_back({x, remain_y});
        stack.push_back({remain_x, y});
        stack.push_back({0, remain_y});
        stack.push_back({remain_x, 0});
        stack.push_back({std::max(remain_x - (y - remain_y), 0), std::min(remain_y + remain_x, y)});
        stack.push_back({std::min(remain_x + remain_y, x), std::max(remain_y - (x - remain_x), 0)})
    }

    return false
}

https://leetcode.cn/problems/perfect-squares/

给定一个数n，找到几个完全平方数(需要最少的个数)，使得其和为n

func numSquares(n int) int {
    f := make([]int, n+1)
    for i := 1; i <= n; i++ {
        minn := math.MaxInt32
        for j := 1; j*j <= i; j++ {
            minn = min(minn, f[i-j*j])
        }
        f[i] = minn + 1
    }
    return f[n]
}

func min(a, b int) int {
    if a < b {
        return a
    }
    return b
}

#include <algorithm>
#include <limits>
#include <vector>

int numSquares(int n) {
    std::vector<int> f(n+1);
    for (int i = 1; i <= n; i++) {
        int minn = std::numeric_limits<int>::max();
        for (int j = 1; j*j <= i; j++) {
            minn = std::min(minn, f[i - j*j]);
        }
        f[i] = minn + 1;
    }
    return f[n];
}

新21点

https://leetcode.cn/problems/new-21-game/

每次从[1, maxPts]范围内随机获得一个整数作为分数累计，直到总分大于k停止，求爱丽丝的分数不超过n的概率是多少

func new21Game(n int, k int, maxPts int) float64 {
    if k == 0 {
        return 1.0
    }
    dp := make([]float64, k + maxPts)
    for i := k; i <= n && i < k + maxPts; i++ {
        dp[i] = 1.0
    }
    for i := k - 1; i >= 0; i-- {
        for j := 1; j <= maxPts; j++ {
            dp[i] += dp[i + j] / float64(maxPts)
        }
    }
    return dp[0]
}

dp[i] += dp[i + j] / float64(maxPts)1和0，相当于把满足条件的个数相加然后除以总数

#include <vector>

double new21Game(int n, int k, int maxPts) {
    std::vector<double> dp(k + maxPts);
    for (int i = k; i <= n && i < k + maxPts; i++) {
        dp[i] = 1.0;
    }
    for (int i = k - 1; i >= 0; i--) {
        for (int j = 1; j <= maxPts; j++) {
            dp[i] += dp[i+j] / double(maxPts);
        }
    }
    return dp[0];
}
