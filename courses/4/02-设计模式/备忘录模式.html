备忘录模式是一种行为设计模式，它可以让你保存一个对象的状态，并在以后恢复它。这样你就不用暴露对象的内部结构，也不用影响它的封装性。

go

// InputText 用于保存数据
type InputText struct {
	content string
}

// Append 追加数据
func (in *InputText) Append(content string) {
	in.content += content
}

// GetText 获取数据
func (in *InputText) GetText() string {
	return in.content
}

// Snapshot 创建快照
func (in *InputText) Snapshot() *Snapshot {
	return &Snapshot{content: in.content}
}

// Restore 从快照中恢复
func (in *InputText) Restore(s *Snapshot) {
	in.content = s.GetText()
}

// Snapshot 快照，用于存储数据快照
// 对于快照来说，只能不能被外部（不同包）修改，只能获取数据，满足封装的特性
type Snapshot struct {
	content string
}

// GetText GetText
func (s *Snapshot) GetText() string {
	return s.content
}

c++

实现了一个简单的编辑器，可以保存和撤销文本的修改

#include <iostream>
#include <string>
#include <stack>
using namespace std;

// 备忘录类，用于存储文本的状态
class Memento {
private:
    string text; // 文本内容
    int cursor; // 光标位置
public:
    // 构造函数，根据给定的文本和光标创建备忘录
    Memento(string text, int cursor) {
        this->text = text;
        this->cursor = cursor;
    }
    // 获取文本内容
    string getText() {
        return text;
    }
    // 获取光标位置
    int getCursor() {
        return cursor;
    }
};

// 编辑器类，用于编辑文本
class Editor {
private:
    string text; // 文本内容
    int cursor; // 光标位置
public:
    // 构造函数，初始化文本和光标
    Editor() {
        text = "";
        cursor = 0;
    }
    // 在光标处插入字符
    void insert(char c) {
        text.insert(cursor, 1, c);
        cursor++;
    }
    // 在光标处删除字符
    void del() {
        if (cursor > 0) {
            text.erase(cursor - 1, 1);
            cursor--;
        }
    }
    // 移动光标
    void move(int offset) {
        cursor += offset;
        if (cursor < 0) {
            cursor = 0;
        }
        if (cursor > text.length()) {
            cursor = text.length();
        }
    }
    // 创建备忘录，保存当前状态
    Memento* createMemento() {
        return new Memento(text, cursor);
    }
    // 恢复备忘录，回到之前的状态
    void restoreMemento(Memento* memento) {
        text = memento->getText();
        cursor = memento->getCursor();
    }
    // 显示文本和光标
    void show() {
        cout << text << endl;
        for (int i = 0; i < cursor; i++) {
            cout << " ";
        }
        cout << "^" << endl;
    }
};

// 管理者类，用于保存和恢复备忘录
class Caretaker {
private:
    stack<Memento*> mementos; // 用栈来存储备忘录
public:
    // 保存备忘录
    void save(Memento* memento) {
        mementos.push(memento);
    }
    // 恢复备忘录
    Memento* undo() {
        if (mementos.empty()) {
            return nullptr;
        }
        Memento* memento = mementos.top();
        mementos.pop();
        return memento;
    }
};

int main() {
    Editor* editor = new Editor(); // 创建编辑器对象
    Caretaker* caretaker = new Caretaker(); // 创建管理者对象

    editor->insert('H'); // 插入H
    editor->insert('e'); // 插入e
    editor->insert('l'); // 插入l
    editor->insert('l'); // 插入l
    editor->insert('o'); // 插入o

    caretaker->save(editor->createMemento()); // 保存当前状态

    editor->show(); // 显示文本和光标

    editor->move(-2); // 光标左移两位
    editor->del(); // 删除一个字符

    caretaker->save(editor->createMemento()); // 保存当前状态

    editor->show(); // 显示文本和光标

    editor->restoreMemento(caretaker->undo()); // 撤销上一次操作

    editor->show(); // 显示文本和光标

    editor->restoreMemento(caretaker->undo()); // 撤销上一次操作

    editor->show(); // 显示文本和光标

    delete editor; // 释放编辑器对象
    delete caretaker; // 释放管理者对象
    return 0;
}
