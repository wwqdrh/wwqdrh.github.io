class Node:
    def __init__(self, data):
        self.data = data
        self.left = None
        self.right = None

class MyTree:
    '''
    @method add_node 从根节点开始遍历直到某个左右节点为空，大的放右边，小的放左边
    @method __iter__ __next__ 实现了迭代器协议，用于迭代节点
    @method getgenerator generator版本，用于简化__iter__ __next__
    '''
    def __init__(self, root):
        self.root = root
    def __iter__(self):
        if self.root is None:
            self.stack = []
        else:
            self.stack = [self.root]
            current = self.root
            while current.left is not None:
                current = current.left
                self.stack.append(current)
        return self
    def __next__(self):
        if len(self.stack) <= 0:
            raise StopIteration
        while len(self.stack) > 0:
            current = self.stack.pop()
            data = current.data
            if current.right is not None:
                current = current.right
                self.stack.append(current)
                while current.left is not None:
                    current = current.left
                    self.stack.append(current)
            return data
        raise StopIteration
    def add_node(self, node):
        current = self.root
        while True:
            if node.data <= current.data:
                if current.left is None:
                    current.left = node
                    return
                else:
                    current = current.left
            else:
                if current.right is None:
                    current.right = node
                    return
                else:
                    current = current.right
    def getgenerator(self):
        if self.root is None:
            self.stack = []
        else:
            self.stack = [self.root]
            current = self.root
            while current.left is not None:
                current = current.left
                self.stack.append(current)
        
        if len(self.stack) <= 0:
            # generator 函数不应该抛出StopIteration
            return

        while len(self.stack) > 0:
            current = self.stack.pop()
            data = current.data
            if current.right is not None:
                current = current.right
                self.stack.append(current)
                while current.left is not None:
                    current = current.left
                    self.stack.append(current)
            yield data
        

if __name__ == "__main__":
    tree = MyTree(Node(16))
    tree.add_node(Node(8))
    tree.add_node(Node(1))
    tree.add_node(Node(17))
    tree.add_node(Node(13))
    tree.add_node(Node(14))
    tree.add_node(Node(9))
    tree.add_node(Node(10))
    tree.add_node(Node(11))

    for i in tree.getgenerator():
        print(i)

// Iterator 迭代器接口
type Iterator interface {
	HasNext() bool
	Next()
	// 获取当前元素，由于 Go 1.15 中还没有泛型，所以我们直接返回 interface{}
	CurrentItem() interface{}
}

// ArrayInt 数组
type ArrayInt []int

// Iterator 返回迭代器
func (a ArrayInt) Iterator() Iterator {
	return &ArrayIntIterator{
		arrayInt: a,
		index:    0,
	}
}

// ArrayIntIterator 数组迭代
type ArrayIntIterator struct {
	arrayInt ArrayInt
	index    int
}

// HasNext 是否有下一个
func (iter *ArrayIntIterator) HasNext() bool {
	return iter.index < len(iter.arrayInt)-1
}

// Next 游标加一
func (iter *ArrayIntIterator) Next() {
	iter.index++
}

// CurrentItem 获取当前元素
func (iter *ArrayIntIterator) CurrentItem() interface{} {
	return iter.arrayInt[iter.index]
}

c++

// 抽象迭代器接口
class Iterator {
public:
  virtual ~Iterator() {}
  virtual std::string First() = 0; // 返回第一个元素
  virtual std::string Next() = 0; // 返回下一个元素
  virtual std::string Current() = 0; // 返回当前元素
  virtual bool IsEnd() = 0; // 判断是否到达末尾
};

// 抽象容器接口
class Container {
public:
  virtual ~Container() {}
  virtual Iterator* CreateIterator() = 0; // 创建迭代器
  virtual int Count() = 0; // 返回元素个数
  virtual void Push(std::string) = 0; // 添加元素
  virtual std::string Pop(int) = 0; // 获取元素
};

// 具体迭代器类
class ConcreteIterator : public Iterator {
public:
  ConcreteIterator(Container* container) : container_(container), index_(0) {} // 构造函数，传入一个容器对象
  ~ConcreteIterator() {} // 析构函数
  std::string First() override { return container_->Pop(0); } // 返回第一个元素
  std::string Next() override { // 返回下一个元素
    std::string str;
    if (index_ < container_->Count() - 1) {
      index_++;
      str = container_->Pop(index_);
    }
    return str;
  }
  std::string Current() override { return container_->Pop(index_); } // 返回当前元素
  bool IsEnd() override { return (index_ == container_->Count() - 1); } // 判断是否到达末尾
private:
  Container* container_; // 容器对象指针
  int index_; // 当前索引
};

// 具体容器类
class ConcreteContainer : public Container {
public:
  ConcreteContainer() : Container(), iterator_(NULL) {} // 构造函数
  ~ConcreteContainer() {} // 析构函数
  Iterator* CreateIterator() override { // 创建迭代器
    if (iterator_ == NULL) iterator_ = new ConcreteIterator(this);
    return iterator_;
  }
  int Count() override { return vec_.size(); } // 返回元素个数
  void Push(std::string s) override { vec_.push_back(s); } // 添加元素
  std::string Pop(int i) override { // 获取元素
    std::string str;
    if (i >= 0 && i < vec_.size()) {
      str = vec_[i];
    }
    return str;
  }
private:
  std::vector<std::string> vec_; // 存储元素的向量
  Iterator* iterator_; // 迭代器对象指针
};

// 客户端代码
int main() {
  // 创建一个具体容器对象，并添加一些元素
  Container* container = new ConcreteContainer();
  container->Push("Tom");
  container->Push("John");
  container->Push("Jack");
  
  // 创建一个具体迭代器对象，并遍历容器中的元素
  Iterator* iterator = container->CreateIterator();
  std::cout << "First element: " << iterator->First() << std::endl;
  while (!iterator->IsEnd()) {
    std::cout << "Next element: " << iterator->Next() << std::endl;
  }
  
  // 释放对象
  delete iterator;
  delete container;
  
  return 0;
}
