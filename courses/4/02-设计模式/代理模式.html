import time

from functools import wraps


class RawCalculator:
    def fib(self, n):
        if n < 2:
            return 1
        return self.fib(n - 2) + self.fib(n - 1)

def memoize(fn):
    __cache = {}

    @wraps(fn)
    def memoized(*args):
        key = fn.__name__, args
        if key in __cache:
            return __cache[key]
        
        __cache[key] = fn(*args)
        return __cache[key]
    
    return memoized

class CalculatorProxy:
    def __init__(self, target):
        self.target = target
        fib = getattr(self.target, 'fib')
        setattr(self.target, 'fib', memoize(fib))
    def __getattr__(self, name):
        return getattr(self.target, name)

if __name__ == "__main__":
    calculator = CalculatorProxy(RawCalculator())

    start_time = time.time()
    fib_sequence = [calculator.fib(x) for x in range(0, 80)]
    end_time = time.time()

    print('calculation the list of {} fib numbers tooks {} seconds'.format(len(fib_sequence), end_time-start_time))

// IUser IUser
type IProxyUser interface {
	Login(username, password string) error
}

// User 用户
type ProxyUser struct {
}

// Login 用户登录
func (u *ProxyUser) Login(username, password string) error {
	// 不实现细节
	return nil
}

// UserProxy 代理类
type ProxyUserProxy struct {
	user *ProxyUser
}

// NewUserProxy NewUserProxy
func NewProxyUserProxy(user *ProxyUser) *ProxyUserProxy {
	return &ProxyUserProxy{
		user: user,
	}
}

// Login 登录，和 user 实现相同的接口
func (p *ProxyUserProxy) Login(username, password string) error {
	// before 这里可能会有一些统计的逻辑
	start := time.Now()

	// 这里是原有的业务逻辑
	if err := p.user.Login(username, password); err != nil {
		return err
	}

	// after 这里可能也有一些监控统计的逻辑
	log.Printf("user login cost time: %s", time.Now().Sub(start))

	return nil
}

// 抽象主题类
class Subject {
public:
  virtual ~Subject() {}
  virtual void Request() = 0; // 公共方法
};

// 真实主题类
class RealSubject : public Subject {
public:
  void Request() override {
    std::cout << "真实主题类处理请求" << std::endl;
  }
};

// 代理主题类
class Proxy : public Subject {
public:
  Proxy(RealSubject* real_subject) : real_subject_(real_subject) {} // 构造函数，传入一个真实主题类的对象
  ~Proxy() { delete real_subject_; } // 析构函数，释放真实主题类的对象
  void PreRequest() { // 在调用真实主题类之前执行的操作
    std::cout << "代理主题类在请求之前做一些事情" << std::endl;
  }
  void Request() override { // 调用真实主题类的方法
    PreRequest();
    real_subject_->Request();
    PostRequest();
  }
  void PostRequest() { // 在调用真实主题类之后执行的操作
    std::cout << "代理主题类在请求之后做一些事情" << std::endl;
  }
private:
  RealSubject* real_subject_; // 真实主题类的对象指针
};

// 客户端代码
int main() {
  // 创建一个真实主题类的对象
  RealSubject* real_subject = new RealSubject();
  
  // 创建一个代理主题类的对象，并传入真实主题类的对象
  Proxy* proxy = new Proxy(real_subject);
  
  // 调用代理主题类的方法
  proxy->Request(); // 输出：
                     // 代理主题类在请求之前做一些事情
                     // 真实主题类处理请求
                     // 代理主题类在请求之后做一些事情
  
  // 释放代理主题类的对象
  delete proxy;
  
  return 0;
}
