type OrderedMap struct {
    keys   []string
    values map[string]interface{}
}

func (m *OrderedMap) Add(key string, value interface{}) {
    m.values[key] = value
    m.keys = append(m.keys, key)
}

func (m *OrderedMap) Get(key string) (interface{}, bool) {
    value, ok := m.values[key]
    return value, ok
}

func (m *OrderedMap) Delete(key string) {
    _, ok := m.Get(key)
    if ok {
        delete(m.values, key)
        for i, k := range m.keys {
            if k == key {
                m.keys = append(m.keys[:i], m.keys[i+1:]...)
                break
            }
        }
    }
}

func (m *OrderedMap) Keys() []string {
    return m.keys
}

func (m *OrderedMap) Values() []interface{} {
    var values []interface{}
    for _, key := range m.keys {
        values = append(values, m.values[key])
    }
    return values
}

4、实现一个LRU缓存

import "container/list"

type LRUCache struct {
    capacity int
    cache    map[int]*list.Element
    list     *list.List
}

type entry struct {
    key   int
    value int
}

func Constructor(capacity int) LRUCache {
    return LRUCache{
        capacity: capacity,
        cache:    make(map[int]*list.Element),
        list:     list.New(),
    }
}

func (this *LRUCache) Get(key int) int {
    if elem, ok := this.cache[key]; ok {
        this.list.MoveToFront(elem)
        return elem.Value.(*entry).value
    }
    return -1
}

func (this *LRUCache) Put(key int, value int) {
    if elem, ok := this.cache[key]; ok {
        elem.Value.(*entry).value = value
        this.list.MoveToFront(elem)
        return
    }
    if this.list.Len() == this.capacity {
        lastElem := this.list.Back()
        delete(this.cache, lastElem.Value.(*entry).key)
        this.list.Remove(lastElem)
    }
    newElem := this.list.PushFront(&entry{key: key, value: value})
    this.cache[key] = newElem
}

6、用栈实现双向队列

type Deque struct {
    head []int
	tail []int
}

func NewDeque() *Deque {
    return &Deque{
        head: make([]int, 0),
        tail: make([]int, 0),
    }
}

func (q *Deque) PushBack(val int) {
    q.tail = append(q.tail, val)
}

func (q *Deque) PushFront(val int) {
    q.head = append(q.head, val)
}

func (q *Deque) PopBack() (val int, ok bool) {
    if len(q.tail) == 0 && len(q.head) == 0 {
        return 0, false
    }

    if len(q.tail) > 0 {
        val = q.tail[len(q.tail)-1]
        q.tail = q.tail[:len(q.tail)-1]
        return val, true
    }

    val = q.head[0]
    q.head = q.head[1:]
    return val, true
}

func (q *Deque) PopFront() (val int, ok bool) {
    if len(q.tail) == 0 && len(q.head) == 0 {
        return 0, false
    }

    if len(q.head) > 0 {
        val = q.head[len(q.head)-1]
        q.head = q.head[:len(q.head)-1]
, true
    }

    val = q.tail[0]
    q.tail = q.tail[1:]
    return val, true
}

func (q *Deque) Size() int {
    return len(q.head) + len(q.tail)
}

func (q *Deque) Front() (val int, ok bool) {
    if len(q.head) > 0 {
        return q.head[len(q.head)-1], true
    }

    if len(q.tail) > 0 {
        return q.tail[0], true
    }

    return 0, false
}

func (q *Deque) Back() (val int, ok bool) {
    if len(q.tail) > 0 {
        return q.tail[len(q.tail)-1], true
    }

    if len(q.head) > 0 {
        return q.head[0], true
    }

    return 0, false
}

2、设计一个短链接服务

主要包含长链接向短链接的映射，其中包含的点在于哈希函数，以及哈希冲突的判断以及解决

解决哈希冲突的方法包括链式法和开放地址法，不过这里链式法无法解决长链和短链一一对应的问题，需要使用开放地址法

长链进来之后先判断是否已经有过记录，没有的话就生成短链之后在数据库中寻找是否已经存在这个短链，如果存在就使用开放地址法重新生成，并且将地址的映射关系加到数据库中。

当来了一个短链接后，后端服务将其解析成长链接

package main

import (
	"crypto/md5"
	"database/sql"
	"encoding/hex"
	"errors"
	"log"
	"sync"

	"github.com/go-redis/redis/v8"
	"gorm.io/driver/mysql"
	"gorm.io/gorm"
)

const (
	// 生成的短链接长度
	shortUrlLength = 6
)

// UrlRecord 定义短链记录模型
type UrlRecord struct {
	ID        uint64 `gorm:"primaryKey"`
	HashValue uint64 `gorm:"not null"`
	ShortUrl  string `gorm:"unique;not null"`
	LongUrl   string `gorm:"not null"`
}

// ShortGenerate 结构体定义
type ShortGenerate struct {
	db  *gorm.DB
	rdb *redis.Client
	mu  sync.Mutex
}

// NewShortGenerate 初始化 ShortGenerate 结构体
func NewShortGenerate(dsn string, redisOptions *redis.Options) (*ShortGenerate, error) {
	db, err := gorm.Open(mysql.Open(dsn), &gorm.Config{})
	if err != nil {
		return nil, err
	}
	rdb := redis.NewClient(redisOptions)
	// 检查 redis 是否连通
	_, redisErr := rdb.Ping(ctx).Result()
	if redisErr != nil {
		return nil, redisErr
	}
	// 自动创建表结构
	err = db.AutoMigrate(&UrlRecord{})
	if err != nil {
		return nil, err
	}

	return &ShortGenerate{
		db:  db,
		rdb: rdb,
	}, nil
}

// Add 生成短链接
func (sg *ShortGenerate) Add(longUrl string) (string, error) {
	// 先从redis中查找是否存在
	shortUrl, err := sg.rdb.Get(ctx, longUrl).Result()
	if err != nil && err != redis.Nil {
		return "", err
	}
	if shortUrl != "" {
		return shortUrl, nil
	}

	// 没有找到，需要计算shortUrl，使用hash函数生成短链接，并且需要判断生成的短链接是否在数据库存在，必须使用开放地址法解决哈希冲突
	hashValue := sg.hash(longUrl)
	shortUrl = sg.shortUrlFromHash(hashValue)

	for i := 0; i < 10; i++ {
		// 使用开放地址法解决哈希冲突
		records := sg.queryRecordByHash(hashValue)
		if len(records) == 0 {
			sg.saveShortUrl(hashValue, shortUrl, longUrl)
			sg.rdb.Set(ctx, longUrl, shortUrl, 0)
			return shortUrl, nil
		}
		if len(records) == 1 && records[0].ShortUrl == shortUrl {
			return shortUrl, nil
		}
		hashValue = sg.nextHashValue(hashValue)
		shortUrl = sg.shortUrlFromHash(hashValue)
	}

	return "", errors.New("failed to generate short url")
}

// Get 通过短链接查询长链接
func (sg *ShortGenerate) Get(shortUrl string) (string, error) {
	var record UrlRecord
	err := sg.db.Where("short_url = ?", shortUrl).First(&record).Error
	if err != nil {
		if errors.Is(err, gorm.ErrRecordNotFound) {
			return "", nil
		}
		return "", err
	}
	return record.LongUrl, nil
}

// 将短链接、长链接、哈希值存储在数据库中
func (sg *ShortGenerate) saveShortUrl(hashValue uint64, shortUrl string, longUrl string) {
	sg.mu.Lock()
	defer sg.mu.Unlock()

	record := UrlRecord{HashValue: hashValue, ShortUrl: shortUrl, LongUrl: longUrl}
	result := sg.db.Create(&record)
	if result.Error != nil {
		log.Fatalln(result.Error)
	}
}

// 通过哈希值查询对应的记录
func (sg *ShortGenerate) queryRecordByHash(hashValue uint64) []UrlRecord {
	var records []UrlRecord
	err := sg.db.Where("hash_value = ?", hashValue).Find(&records).Error
	if err != nil && !errors.Is(err, gorm.ErrRecordNotFound) {
		log.Fatalln(err)
	}
	return records
}

// 哈希函数，返回uint64类型的哈希值
func (sg *ShortGenerate) hash(longUrl string) uint64 {
	h := md5.New()
	h.Write([]byte(longUrl))
	return uint64(h.Sum32())
}

// 使用开放地址法解决哈希冲突
func (sg *ShortGenerate) nextHashValue(hashValue uint64) uint64 {
	return hashValue*101 % 1023 // 假设桶的大小为1023
}

// 根据哈希值生成短链接字符串
func (sg *ShortGenerate) shortUrlFromHash(hashValue uint64) string {
	hexHash := hex.EncodeToString([]byte{byte(hashValue >> 24), byte(hashValue >> 16), byte(hashValue >> 8), byte(hashValue)})
	return hexHash[:shortUrlLength]
}
